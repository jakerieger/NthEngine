32 bool RenderTarget::Initialize() {
33 if (mInitialized) {
34 Log::Warn("RenderTarget", "Already initialized");
35 return true;
36 }
37
38 // Initialize render context
39 if (!mContext.Initialize(mWidth, mHeight)) {
40 Log::Error("RenderTarget", "Failed to initialize render context");
41 return false;
42 }
43
44 // Create framebuffer if not rendering to window
45 if (mType == RenderTargetType::Framebuffer) {
46 CreateFramebuffer();
47 }
48
49 mInitialized = true;
50 Log::Debug("RenderTarget",
51 "Initialized {} render target ({}x{})",
52 mType == RenderTargetType::Window ? "window" : "framebuffer",
53 mWidth,
54 mHeight);
55
56 return true;
57 }

59 void RenderTarget::Shutdown() {

69 mInitialized = false;

88 void RenderTarget::Resize(u32 width, u32 height) {

106 void RenderTarget::CreateFramebuffer() {
107 ASTERA_ASSERT(mType == RenderTargetType::Framebuffer);
108
109 // Create framebuffer
110 GLCall(glGenFramebuffers, 1, &mFramebufferID);
111 GLCall(glBindFramebuffer, GL_FRAMEBUFFER, mFramebufferID);
112
113 // Create color texture
114 GLCall(glGenTextures, 1, &mTextureID);
115 GLCall(glBindTexture, GL_TEXTURE_2D, mTextureID);
116 GLCall(glTexImage2D,
117 GL_TEXTURE_2D,
118 0,
119 GL_RGBA8,
120 (GLsizei)mWidth,
121 (GLsizei)mHeight,
122 0,
123 GL_RGBA,
124 GL_UNSIGNED_BYTE,
125 nullptr);
126 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
127 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
128 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
129 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
130 GLCall(glFramebufferTexture2D, GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, mTextureID, 0);
131
132 // Create depth/stencil buffer if requested
133 if (mEnableDepth || mEnableStencil) {
134 GLCall(glGenRenderbuffers, 1, &mDepthStencilID);
135 GLCall(glBindRenderbuffer, GL_RENDERBUFFER, mDepthStencilID);
136
137 GLenum format;
138 GLenum attachment;
139
140 if (mEnableDepth && mEnableStencil) {
141 format = GL_DEPTH24_STENCIL8;
142 attachment = GL_DEPTH_STENCIL_ATTACHMENT;
143 } else if (mEnableDepth) {
144 format = GL_DEPTH_COMPONENT24;
145 attachment = GL_DEPTH_ATTACHMENT;
146 } else {
147 format = GL_STENCIL_INDEX8;
148 attachment = GL_STENCIL_ATTACHMENT;
149 }
150
151 GLCall(glRenderbufferStorage, GL_RENDERBUFFER, format, (GLsizei)mWidth, (GLsizei)mHeight);
152 GLCall(glFramebufferRenderbuffer, GL_FRAMEBUFFER, attachment, GL_RENDERBUFFER, mDepthStencilID);
153 }
154
155 // Check framebuffer completeness
156 GLenum status = GLCall(glCheckFramebufferStatus, GL_FRAMEBUFFER);
157 if (status != GL_FRAMEBUFFER_COMPLETE) {
158 Log::Error("RenderTarget", "Framebuffer is not complete: {}", status);
159 }
160
161 GLCall(glBindFramebuffer, GL_FRAMEBUFFER, 0);
162 Log::Debug("RenderTarget", "Created framebuffer ({}x{})", mWidth, mHeight);
163 }

165 void RenderTarget::DestroyFramebuffer() {
166 if (mFramebufferID != 0) {
167 GLCall(glDeleteFramebuffers, 1, &mFramebufferID);
168 mFramebufferID = 0;
169 }
170
171 if (mTextureID != 0) {
172 GLCall(glDeleteTextures, 1, &mTextureID);
173 mTextureID = 0;
174 }
175
176 if (mDepthStencilID != 0) {
177 GLCall(glDeleteRenderbuffers, 1, &mDepthStencilID);
178 mDepthStencilID = 0;
179 }
