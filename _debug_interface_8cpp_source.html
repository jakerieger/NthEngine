37 void DebugManager::AttachOverlay(const std::string& name, IDebugOverlay* overlay) {
38 mOverlays.insert_or_assign(name, Overlay {overlay, true});

41 void DebugManager::DetachOverlays() {
42 for (auto [name, overlay] : mOverlays) {
43 overlay.overlay = nullptr;
44 overlay.enabled = false;
45 }
46 mOverlays.clear();
47 }

49 void DebugManager::SetOverlayEnabled(const std::string& name, bool enabled) {
50 const auto it = mOverlays.find(name);
51 if (it != mOverlays.end()) { it->second.enabled = enabled; }
52 }
53
54 bool DebugManager::GetOverlayEnabled(const std::string& name) const {
55 return mOverlays.at(name).enabled;
56 }

58 void DebugManager::Update(float deltaTime) const {
59 for (const auto overlay : mOverlays | std::views::values) {
60 overlay.overlay->OnUpdate(deltaTime);
61 }
62 }
63
64 void DebugManager::Render() const {

66 if (overlay.enabled) { overlay.overlay->OnRender(); }
67 }
68 }

70 void DebugManager::HandleEvent(const Event& event) const {
