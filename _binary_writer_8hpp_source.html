197 template<typename T>
198 T ConvertEndian(T value) {
199 if (mEndian == Endian::Native) {
200 return value;
201 }
202
203 bool needSwap =
204 (mEndian == Endian::Little && !IsLittleEndian()) || (mEndian == Endian::Big && IsLittleEndian());
205
206 if (!needSwap || sizeof(T) == 1) {
207 return value;
208 }
209
210 return SwapBytes(value);
211 }
212
214 template<typename T>
215 T SwapBytes(T value) {
216 static_assert(std::is_fundamental<T>::value, "SwapBytes only works with fundamental types");
217
218 union {
219 T value;
220 u8 bytes[sizeof(T)];
221 } src, dst;
222
223 src.value = value;
224
225 for (size_t i = 0; i < sizeof(T); ++i) {
226 dst.bytes[i] = src.bytes[sizeof(T) - 1 - i];
227 }
228
229 return dst.value;
230 }

233 template<typename T>
