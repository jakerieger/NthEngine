36 Result<vector<u8>> AssetManager::GetAssetData(AssetID id) {
37 if (!sInitialized) {
38 Initialize();
39 }
40
41 if (!sAssetPaths.contains(id)) {
42 return unexpected(fmt::format("Asset with given ID `{}` not found", id));
43 }
44
45 const auto assetPath = sAssetPaths[id];
46 if (!exists(assetPath)) {
47 return unexpected(fmt::format("Asset `{}` does not exist", assetPath.string()));
48 }
49
50 auto readResult = IO::ReadBytes(assetPath);
51 if (readResult.has_value()) {
52 return *readResult;
53 } else {
54 return unexpected(fmt::format("Failed to read asset `{}`", assetPath.string()));
55 }
56 }

58 bool AssetManager::Initialize() {
59 if (!exists(sWorkingDirectory)) {
60 Log::Error("AssetManager", "Working directory does not exist");
61 return false;
62 }
63
64 // Iterate over working directory looking for ".asset" files
65 for (const auto& entry : fs::recursive_directory_iterator(sWorkingDirectory)) {
66 if (entry.is_regular_file() && entry.path().extension() == ".asset") {
67 const auto extLen = string(".asset").size();
68 auto srcFilename =
69 entry.path().filename().string().substr(0, entry.path().filename().string().size() - extLen);
70 const auto assetPath = entry.path().parent_path() / srcFilename;
71 if (!exists(assetPath)) {
72 Log::Error("AssetManager", "Asset file `{}` does not exist", assetPath.string());
73 continue;
74 }
75
76 // Read asset ID from descriptor
77 pugi::xml_document doc;
78 doc.load_file(entry.path().string().c_str());
79 const AssetID id = doc.child("Asset").attribute("id").as_ullong();
80
81 sAssetPaths[id] = assetPath;
82
83 Log::Info("AssetManager",
84 "Found asset `{}` with ID {} and type {}",
85 assetPath.string(),
86 id,
87 (u32)AssetTypeFromID(id));
88 }
89 }
90
91 sInitialized = true;
92 return true;
93 }

95 void AssetManager::Reload() {
96 sInitialized = false;

98 Initialize();
