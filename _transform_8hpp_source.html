38
39 void Translate(const Vec2& translation) {

41 }
42
43 void Rotate(f32 angle) {
44 rotation.x += angle;
45 }
46
47 void Scale(const Vec2& scaleFactor) {
48 scale *= scaleFactor;
49 }
50
51 N_ND Mat4 GetMatrix() const {
52 auto mat = Mat4(1.0f);
53 mat = glm::translate(mat, glm::vec3(position.x, position.y, 0.0f));
54 mat = glm::rotate(mat, glm::radians(rotation.x), glm::vec3(0.0f, 0.0f, 1.0f)); // Z-axis rotation for 2D
55 mat = glm::scale(mat, glm::vec3(scale.x, scale.y, 1.0f));
56 return mat;
57 }
