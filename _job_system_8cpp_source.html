31 void JobSystem::Initialize(size_t numThreads) {
32 if (mInitialized.exchange(true)) {
33 Log::Warn("JobSystem", "Already initialized");
34 return;
35 }
36
37 // Use hardware concurrency if not specified
38 if (numThreads == 0) {
39 numThreads = std::thread::hardware_concurrency();
40 if (numThreads == 0)
41 numThreads = 4; // Fallback
42 }
43
44 mShutdown = false;
45 mWorkers.resize(numThreads);
46
47 // Create worker threads
48 for (size_t i = 0; i < numThreads; ++i) {
49 mWorkers[i].thread = std::thread([this, i]() { WorkerLoop(i); });
50
51 // Store thread ID for GetCurrentWorkerID
52 {
53 std::lock_guard<std::mutex> lock(mThreadMapMutex);
54 mWorkers[i].threadId = mWorkers[i].thread.get_id();
55 mThreadIdToWorkerIndex[mWorkers[i].threadId] = i;
56 }
57 }
58
59 Log::Info("JobSystem", "Initialized with {} worker threads", numThreads);
60 }

62 void JobSystem::Shutdown() {
63 if (!mInitialized.exchange(false)) {
64 return;
65 }
66
67 Log::Info("JobSystem", "Shutting down job system...");
68
69 // Signal shutdown
70 mShutdown = true;
71
72 // Wake up all workers
73 mGlobalCondition.notify_all();
74
75 // Join all worker threads
76 for (auto& worker : mWorkers) {
77 if (worker.thread.joinable()) {
78 worker.thread.join();
79 }
80 }
81
82 // Clear worker data
83 mWorkers.clear();
84
85 {
86 std::lock_guard<std::mutex> lock(mThreadMapMutex);
87 mThreadIdToWorkerIndex.clear();
88 }
89
90 // Clear any remaining jobs
91 {
92 std::lock_guard<std::mutex> lock(mGlobalMutex);
93 while (!mGlobalQueue.empty()) {
94 mGlobalQueue.pop();
95 }
96 }
97
98 Log::Info("JobSystem",
99 "Shutdown complete. Total jobs: submitted={}, completed={}",
100 mTotalJobsSubmitted.load(),
101 mTotalJobsCompleted.load());
102 }

104 void JobSystem::Submit(Job job) {
105 if (!mInitialized) {
106 Log::Error("JobSystem", "Cannot submit job - system not initialized");
107 return;
108 }
109
110 mTotalJobsSubmitted.fetch_add(1, std::memory_order_relaxed);
111

120 void JobSystem::SubmitToWorker(Job job, size_t workerId) {
121 if (!mInitialized) {
122 Log::Error("JobSystem", "Cannot submit job - system not initialized");
123 return;
124 }
125
126 if (workerId >= mWorkers.size()) {
127 Log::Warn("JobSystem", "Invalid worker ID {}, submitting to global queue", workerId);
128 Submit(std::move(job));
129 return;
130 }
131
132 mTotalJobsSubmitted.fetch_add(1, std::memory_order_relaxed);
133
134 {
135 std::lock_guard<std::mutex> lock(mWorkers[workerId].queueMutex);
136 mWorkers[workerId].localQueue.push(std::move(job));
137 }
138
139 mGlobalCondition.notify_one();
140 }

142 shared_ptr<JobSystem::JobCounter> JobSystem::SubmitBatch(const vector<Job>& jobs) {
143 if (jobs.empty()) {
144 return nullptr;
145 }
146
147 auto counter = make_shared<JobCounter>();
148 counter->remaining = static_cast<i32>(jobs.size());
149
150 for (const auto& job : jobs) {
151 Submit([job, counter]() {
152 job();
153 counter->remaining.fetch_sub(1, std::memory_order_release);
154 });
155 }
156
157 return counter;

160 shared_ptr<JobSystem::JobCounter> JobSystem::SubmitBatchChunked(const vector<Job>& jobs, size_t chunkSize) {
161 if (jobs.empty()) {
162 return nullptr;
163 }
164
165 // Auto-determine chunk size
166 if (chunkSize == 0) {
167 chunkSize = Math::Max(size_t(1), jobs.size() / (mWorkers.size() * 2));
168 chunkSize = Math::Min(chunkSize, kDefaultChunkSize);
169 }
170
171 vector<Job> chunkedJobs;
172 for (size_t i = 0; i < jobs.size(); i += chunkSize) {
173 size_t end = Math::Min(i + chunkSize, jobs.size());
174
175 // Create a job that executes a chunk
176 chunkedJobs.push_back([&jobs, i, end]() {
177 for (size_t j = i; j < end; ++j) {
178 jobs[j]();
179 }
180 });
181 }
182
183 return SubmitBatch(chunkedJobs);
184 }

199 bool JobSystem::WaitForCounterTimeout(const shared_ptr<JobCounter>& counter, u32 timeoutMs) {
200 if (!counter)
201 return true;
202
203 auto start = std::chrono::steady_clock::now();
204 auto timeout = std::chrono::milliseconds(timeoutMs);
205
206 while (!counter->IsComplete()) {
207 if (std::chrono::steady_clock::now() - start > timeout) {
208 return false; // Timeout
209 }
210
211 if (!ExecuteNextJob()) {
212 std::this_thread::yield();
213 }
214 }
215
216 return true; // Completed
217 }

230 bool JobSystem::ExecuteNextJob() {
231 if (!mInitialized)
232 return false;
233
234 Job job;
235 i32 workerID = GetCurrentWorkerID();
236
237 if (workerID >= 0) {
238 // We're on a worker thread, try to get a job
239 if (TryGetJob(static_cast<size_t>(workerID), job)) {
240 job();
241 mTotalJobsCompleted.fetch_add(1, std::memory_order_relaxed);
242 return true;
243 }
244 } else {
245 // We're on the main thread or external thread, try global queue
246 std::lock_guard<std::mutex> lock(mGlobalMutex);
247 if (!mGlobalQueue.empty()) {
248 job = std::move(mGlobalQueue.front());
249 mGlobalQueue.pop();
250
251 // Execute without lock
252 lock.~lock_guard();
253 job();
254 mTotalJobsCompleted.fetch_add(1, std::memory_order_relaxed);
255 return true;
256 }
257 }
258
259 return false;
260 }

264 auto it = mThreadIdToWorkerIndex.find(std::this_thread::get_id());

268 JobSystem::Statistics JobSystem::GetStatistics() const {
269 Statistics stats;
270 stats.totalJobsSubmitted = mTotalJobsSubmitted.load(std::memory_order_relaxed);
271 stats.totalJobsCompleted = mTotalJobsCompleted.load(std::memory_order_relaxed);
272
273 {
274 std::lock_guard<std::mutex> lock(mGlobalMutex);
275 stats.jobsInGlobalQueue = mGlobalQueue.size();
276 }
277
278 stats.jobsPerWorker.resize(mWorkers.size());
279 for (size_t i = 0; i < mWorkers.size(); ++i) {
280 std::lock_guard<std::mutex> lock(mWorkers[i].queueMutex);
281 stats.jobsPerWorker[i] = mWorkers[i].localQueue.size();
282 stats.jobsInLocalQueues += stats.jobsPerWorker[i];
283 }
284
285 return stats;
286 }

288 void JobSystem::WorkerLoop(size_t workerId) {
289 while (!mShutdown) {
290 Job job;
291
292 if (TryGetJob(workerId, job)) {
293 // Execute the job
294 job();
295 mTotalJobsCompleted.fetch_add(1, std::memory_order_relaxed);
296 mWorkers[workerId].jobsProcessed.fetch_add(1, std::memory_order_relaxed);
297 } else {
298 // No job found, wait for notification
299 std::unique_lock<std::mutex> lock(mGlobalMutex);
300 mGlobalCondition.wait_for(lock, std::chrono::milliseconds(10), [this]() {
301 return mShutdown.load() || !mGlobalQueue.empty();
302 });
303 }
304 }
305 }

307 bool JobSystem::TryGetJob(size_t workerId, Job& outJob) {
308 // 1. Try local queue first (best cache locality)
309 {
310 std::lock_guard<std::mutex> lock(mWorkers[workerId].queueMutex);
311 if (!mWorkers[workerId].localQueue.empty()) {
312 outJob = std::move(mWorkers[workerId].localQueue.front());
313 mWorkers[workerId].localQueue.pop();
314 return true;
315 }
316 }
317
318 // 2. Try global queue
319 {
320 std::lock_guard<std::mutex> lock(mGlobalMutex);
321 if (!mGlobalQueue.empty()) {
322 outJob = std::move(mGlobalQueue.front());
323 mGlobalQueue.pop();
324 return true;
325 }
326 }
327
328 // 3. Try work stealing from other workers
329 return TryStealJob(workerId, outJob);
330 }

332 bool JobSystem::TryStealJob(size_t thiefId, Job& outJob) {
333 // Random starting point to distribute stealing evenly
334 size_t start = (thiefId + 1) % mWorkers.size();
335 for (size_t i = 0; i < mWorkers.size(); ++i) {
336 size_t victimId = (start + i) % mWorkers.size();
337 if (victimId == thiefId)
338 continue;
339
340 std::lock_guard<std::mutex> lock(mWorkers[victimId].queueMutex);
341 if (!mWorkers[victimId].localQueue.empty()) {
342 outJob = std::move(mWorkers[victimId].localQueue.front());
343 mWorkers[victimId].localQueue.pop();
344 return true;
345 }
346 }
347
348 return false;
