170 void BinaryWriter::WriteRaw(const void* data, size_t size) {
171 const auto byteData = CAST<const u8*>(data);
172
173 if (mWritePos < mBuffer.size()) {
174 // Overwriting existing data
175 const size_t available = mBuffer.size() - mWritePos;
176 const size_t toCopy = Math::Min(size, available);
177 std::memcpy(mBuffer.data() + mWritePos, byteData, toCopy);
178 mWritePos += toCopy;
179 if (toCopy < size) {
180 // Append remaining data
181 mBuffer.insert(mBuffer.end(), byteData + toCopy, byteData + size);
182 mWritePos = mBuffer.size();
183 }
184 } else {
185 // Appending new data
186 mBuffer.insert(mBuffer.end(), byteData, byteData + size);
187 mWritePos = mBuffer.size();
188 }
189 }
