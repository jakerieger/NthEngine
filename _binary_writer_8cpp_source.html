165 void BinaryWriter::WriteRaw(const void* data, size_t size) {
166 const u8* byteData = static_cast<const u8*>(data);
167
168 if (mWritePos < mBuffer.size()) {
169 // Overwriting existing data
170 size_t available = mBuffer.size() - mWritePos;
171 size_t toCopy = std::min(size, available);
172 std::memcpy(mBuffer.data() + mWritePos, byteData, toCopy);
173 mWritePos += toCopy;
174 if (toCopy < size) {
175 // Append remaining data
176 mBuffer.insert(mBuffer.end(), byteData + toCopy, byteData + size);
177 mWritePos = mBuffer.size();
178 }
179 } else {
180 // Appending new data
181 mBuffer.insert(mBuffer.end(), byteData, byteData + size);
182 mWritePos = mBuffer.size();
183 }
184 }
