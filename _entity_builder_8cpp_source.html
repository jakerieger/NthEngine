64 EntityBuilder& EntityBuilder::AddSpriteRenderer(const SpriteRendererDescriptor& descriptor) {
65 // Load texture
66 auto& resourceManager = mScene->GetResourceManager();
67 const auto loadResult = resourceManager.LoadResource<TextureSprite>(descriptor.texture);
68 if (!loadResult) {
69 throw std::runtime_error("Could not load texture sprite");
70 }
71 const ResourceHandle<TextureSprite> spriteHandle =
72 resourceManager.FetchResource<TextureSprite>(descriptor.texture);
73
74 if (!spriteHandle->IsValid()) {
75 throw std::runtime_error("Could not load texture sprite - handle invalid");
76 }
77
78 // Update entity
79 auto& sprite = mScene->GetState().AddComponent<SpriteRenderer>(mEntity);
80 sprite.geometry = Geometry::CreateQuad();
81 sprite.sprite = spriteHandle;
82
83 return *this;
84 }
