71 void Scene::Render(RenderContext& context) {
72 u32 screenWidth = 0, screenHeight = 0;
73 context.GetViewportDimensions(screenWidth, screenHeight);
74
75 const auto iter = mState.View<Transform, SpriteRenderer>().each();
76 for (auto [entity, transform, sprite] : iter) {
77 context.Submit(DrawSpriteCommand {&sprite, &transform, {screenWidth, screenHeight}, {1, 1, 1, 1}});
78 }

81 void Scene::LoadXML(const Path& filename, ScriptEngine& engine) {
82 mState.Reset();
83 mResourceManager.Clear();
84
85 SceneDescriptor descriptor;
86 SceneParser::DeserializeDescriptorXML(filename, descriptor);
87 SceneParser::DescriptorToScene(descriptor, this, engine);
88
89 Log::Debug("Scene", "Loaded scene: `{}`", descriptor.name);
