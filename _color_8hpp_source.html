303 template<>
304 ASTERA_KEEP inline u32 Color::To() const {
305 const auto redByte = CAST<u8>(FloatToU32(mRed));
306 const auto greenByte = CAST<u8>(FloatToU32(mGreen));
307 const auto blueByte = CAST<u8>(FloatToU32(mBlue));
308 const auto alphaByte = CAST<u8>(FloatToU32(mAlpha));
309 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
310 }
311
314 template<>
315 ASTERA_KEEP inline string Color::To() const {
316 const u32 r = U32ToFloat(mRed + 0.5f);
317 const u32 g = U32ToFloat(mGreen + 0.5f);
318 const u32 b = U32ToFloat(mBlue + 0.5f);
319 std::stringstream ss;
320 ss << "#" << std::hex << std::setfill('0') << std::setw(2) << (r & 0xFF) << std::setw(2) << (g & 0xFF)
321 << std::setw(2) << (b & 0xFF);
322 return ss.str();
323 }

327 template<>
