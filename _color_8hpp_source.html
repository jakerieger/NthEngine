278 template<>
279 N_ND inline u32 Color::To() const {
280 const auto redByte = CAST<u8>(FloatToU32(mRed));
281 const auto greenByte = CAST<u8>(FloatToU32(mGreen));
282 const auto blueByte = CAST<u8>(FloatToU32(mBlue));
283 const auto alphaByte = CAST<u8>(FloatToU32(mAlpha));
284 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
285 }
286
289 template<>
290 N_ND inline string Color::To() const {
291 const u32 r = U32ToFloat(mRed + 0.5f);
292 const u32 g = U32ToFloat(mGreen + 0.5f);
293 const u32 b = U32ToFloat(mBlue + 0.5f);
294 std::stringstream ss;
295 ss << "#" << std::hex << std::setfill('0') << std::setw(2) << (r & 0xFF) << std::setw(2) << (g & 0xFF)
296 << std::setw(2) << (b & 0xFF);
297 return ss.str();
298 }
