156 void InputManager::UpdateMousePosition(const f64 x, const f64 y) {
157 if (!mEnabled) return;
158
159 mMouseDeltaX = CAST<f32>(x);
160 mMouseDeltaY = CAST<f32>(y);
161
162 constexpr f32 deadZone = 2.5f; // Might need to tweak this, also frame-rate dependent :(
163 if (std::abs(mMouseDeltaX) < deadZone) mMouseDeltaX = 0.0f;
164 if (std::abs(mMouseDeltaY) < deadZone) mMouseDeltaY = 0.0f;
165
166 mMouseX += x;
167 mMouseY += y;
168 }
