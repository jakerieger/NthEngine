35 bool InputManager::GetAction(const string& action) {
36 if (!mInputMap.has_value() || !mInputMap->actions.contains(action)) {
37 return false;
38 }
39
40 const auto map = mInputMap->actions.at(action);
41 const bool keyDown = std::ranges::any_of(map.keys, [this](u32 key) { return GetKeyDown((i32)key); });
42 const bool mouseDown =
43 std::ranges::any_of(map.mouseButtons, [this](u32 button) { return GetMouseButtonDown((i32)button); });
44
45 return keyDown || mouseDown;
46 }

202 void InputManager::UpdateMousePosition(const f64 x, const f64 y) {
203 if (!mEnabled)
204 return;
205
206 mMouseDeltaX = CAST<f32>(x);
207 mMouseDeltaY = CAST<f32>(y);
208
209 constexpr f32 deadZone = 2.5f; // Might need to tweak this, also frame-rate dependent :(
210 if (std::abs(mMouseDeltaX) < deadZone)
211 mMouseDeltaX = 0.0f;
212 if (std::abs(mMouseDeltaY) < deadZone)
213 mMouseDeltaY = 0.0f;
214
215 mMouseX += x;
216 mMouseY += y;
217 }
