180 void InputManager::UpdateMousePosition(const f64 x, const f64 y) {
181 if (!mEnabled) return;
182
183 mMouseDeltaX = CAST<f32>(x);
184 mMouseDeltaY = CAST<f32>(y);
185
186 constexpr f32 deadZone = 2.5f; // Might need to tweak this, also frame-rate dependent :(
187 if (std::abs(mMouseDeltaX) < deadZone) mMouseDeltaX = 0.0f;
188 if (std::abs(mMouseDeltaY) < deadZone) mMouseDeltaY = 0.0f;
189
190 mMouseX += x;
191 mMouseY += y;
192 }
