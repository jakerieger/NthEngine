200 void Color::ToHSV(f32& h, f32& s, f32& v) const {
201 const f32 max = Math::Max(Math::Max(mRed, mGreen), mBlue);
202 const f32 min = Math::Min(Math::Min(mRed, mGreen), mBlue);
203 const f32 delta = max - min;
204
205 v = max;
206 s = (max > 0.0f) ? (delta / max) : 0.0f;
207
208 if (delta < 0.00001f) {
209 h = 0.0f; // No saturation, so hue is undefined
210 } else {
211 if (mRed >= max) {
212 h = (mGreen - mBlue) / delta;
213 } else if (mGreen >= max) {
214 h = 2.0f + (mBlue - mRed) / delta;
215 } else {
216 h = 4.0f + (mRed - mGreen) / delta;
217 }
218
219 h *= 60.0f; // Convert to degrees
220 if (h < 0.0f) { h += 360.0f; }
221 h /= 360.0f; // Normalize to [0, 1]
222 }
223 }

225 Color Color::FromHSV(f32 h, f32 s, f32 v, f32 a) {
226 if (s <= 0.0f) { return Color(v, a); }
227
228 h = std::fmod(h, 1.0f) * 6.0f; // Map h to [0, 6)
229
230 const int i = CAST<int>(h);
231 const f32 f = h - i;
232 const f32 p = v * (1.0f - s);
233 const f32 q = v * (1.0f - s * f);
234 const f32 t = v * (1.0f - s * (1.0f - f));
235
236 switch (i) {
237 case 0:
238 return {v, t, p, a};
239 case 1:
240 return {q, v, p, a};
241 case 2:
242 return {p, v, t, a};
243 case 3:
244 return {p, q, v, a};
245 case 4:
246 return {t, p, v, a};
247 default:
248 return {v, p, q, a};
249 }
250 }

268 Color Color::AlphaBlend(const Color& foreground, const Color& background) {
269 const f32 alpha = foreground.A();
270 if (alpha == 0.f) { return background; }
271 const f32 invAlpha = 1.f - alpha;
272
273 f32 backAlpha = background.A();
274 if (backAlpha == 1.f) {
275 return {alpha * foreground.R() + invAlpha * background.R(),
276 alpha * foreground.G() + invAlpha * background.G(),
277 alpha * foreground.B() + invAlpha * background.B(),
278 1.f};
279 }
280
281 backAlpha = (backAlpha * invAlpha) / 1.f;
282 const f32 outAlpha = alpha + backAlpha;
283 ASTERA_ASSERT_MSG(outAlpha != 0.f, "outAlpha != 0.0f");
284 return {alpha * foreground.R() + invAlpha * background.R() / outAlpha,
285 alpha * foreground.G() + invAlpha * background.G() / outAlpha,
286 alpha * foreground.B() + invAlpha * background.B() / outAlpha,
287 1.f};
288 }

308 Color Color::Overlay(const Color& a, const Color& b) {
309 auto overlayComponent = [](f32 a, f32 b) -> f32 {
310 if (a < 0.5f) {
311 return 2.0f * a * b;
312 } else {
313 return 1.0f - 2.0f * (1.0f - a) * (1.0f - b);
314 }
315 };
316
317 return {overlayComponent(a.R(), b.R()),
318 overlayComponent(a.G(), b.G()),
319 overlayComponent(a.B(), b.B()),
320 a.A() * b.A()};
321 }

323 Color Color::SoftLight(const Color& a, const Color& b) {
324 auto softLightComponent = [](f32 a, f32 b) -> f32 {
325 if (b < 0.5f) {
326 return a - (1.0f - 2.0f * b) * a * (1.0f - a);
327 } else {
328 f32 d;
329 if (a < 0.25f) {
330 d = ((16.0f * a - 12.0f) * a + 4.0f) * a;
331 } else {
332 d = std::sqrt(a);
333 }
334 return a + (2.0f * b - 1.0f) * (d - a);
335 }
336 };
337
338 return {softLightComponent(a.R(), b.R()),
339 softLightComponent(a.G(), b.G()),
340 softLightComponent(a.B(), b.B()),
341 a.A() * b.A()};
342 }
