176 void Color::ToHSV(f32& h, f32& s, f32& v) const {
177 const f32 max = N_MAX(N_MAX(mRed, mGreen), mBlue);
178 const f32 min = N_MIN(N_MIN(mRed, mGreen), mBlue);
179 const f32 delta = max - min;
180
181 v = max;
182 s = (max > 0.0f) ? (delta / max) : 0.0f;
183
184 if (delta < 0.00001f) {
185 h = 0.0f; // No saturation, so hue is undefined
186 } else {
187 if (mRed >= max) {
188 h = (mGreen - mBlue) / delta;
189 } else if (mGreen >= max) {
190 h = 2.0f + (mBlue - mRed) / delta;
191 } else {
192 h = 4.0f + (mRed - mGreen) / delta;
193 }
194
195 h *= 60.0f; // Convert to degrees
196 if (h < 0.0f) { h += 360.0f; }
197 h /= 360.0f; // Normalize to [0, 1]
198 }
199 }

201 Color Color::FromHSV(f32 h, f32 s, f32 v, f32 a) {
202 if (s <= 0.0f) { return Color(v, a); }
203
204 h = std::fmod(h, 1.0f) * 6.0f; // Map h to [0, 6)
205
206 const int i = CAST<int>(h);
207 const f32 f = h - i;
208 const f32 p = v * (1.0f - s);
209 const f32 q = v * (1.0f - s * f);
210 const f32 t = v * (1.0f - s * (1.0f - f));
211
212 switch (i) {
213 case 0:
214 return {v, t, p, a};
215 case 1:
216 return {q, v, p, a};
217 case 2:
218 return {p, v, t, a};
219 case 3:
220 return {p, q, v, a};
221 case 4:
222 return {t, p, v, a};
223 default:
224 return {v, p, q, a};
225 }
226 }

244 Color Color::AlphaBlend(const Color& foreground, const Color& background) {
245 const f32 alpha = foreground.A();
246 if (alpha == 0.f) { return background; }
247 const f32 invAlpha = 1.f - alpha;
248
249 f32 backAlpha = background.A();
250 if (backAlpha == 1.f) {
251 return {alpha * foreground.R() + invAlpha * background.R(),
252 alpha * foreground.G() + invAlpha * background.G(),
253 alpha * foreground.B() + invAlpha * background.B(),
254 1.f};
255 }
256
257 backAlpha = (backAlpha * invAlpha) / 1.f;
258 const f32 outAlpha = alpha + backAlpha;
259 N_ASSERT_MSG(outAlpha != 0.f, "outAlpha != 0.0f");
260 return {alpha * foreground.R() + invAlpha * background.R() / outAlpha,
261 alpha * foreground.G() + invAlpha * background.G() / outAlpha,
262 alpha * foreground.B() + invAlpha * background.B() / outAlpha,
263 1.f};
264 }

284 Color Color::Overlay(const Color& a, const Color& b) {
285 auto overlayComponent = [](f32 a, f32 b) -> f32 {
286 if (a < 0.5f) {
287 return 2.0f * a * b;
288 } else {
289 return 1.0f - 2.0f * (1.0f - a) * (1.0f - b);
290 }
291 };
292
293 return {overlayComponent(a.R(), b.R()),
294 overlayComponent(a.G(), b.G()),
295 overlayComponent(a.B(), b.B()),
296 a.A() * b.A()};
297 }

299 Color Color::SoftLight(const Color& a, const Color& b) {
300 auto softLightComponent = [](f32 a, f32 b) -> f32 {
301 if (b < 0.5f) {
302 return a - (1.0f - 2.0f * b) * a * (1.0f - a);
303 } else {
304 f32 d;
305 if (a < 0.25f) {
306 d = ((16.0f * a - 12.0f) * a + 4.0f) * a;
307 } else {
308 d = std::sqrt(a);
309 }
310 return a + (2.0f * b - 1.0f) * (d - a);
311 }
312 };
313
314 return {softLightComponent(a.R(), b.R()),
315 softLightComponent(a.G(), b.G()),
316 softLightComponent(a.B(), b.B()),
317 a.A() * b.A()};
318 }
