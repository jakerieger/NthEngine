42 static Rigidbody2DDescriptor ParseRigidbodyComponent(const pugi::xml_node& rigidbodyNode) {
43 Rigidbody2DDescriptor rigidbody {};
44
45 if (const auto node = rigidbodyNode.child("BodyType")) { rigidbody.type = node.child_value(); }
46 if (const auto node = rigidbodyNode.child("Velocity")) {
47 rigidbody.velocity.x = node.attribute("x").as_float();
48 rigidbody.velocity.y = node.attribute("y").as_float();
49 }
50 if (const auto node = rigidbodyNode.child("Acceleration")) {
51 rigidbody.acceleration.x = node.attribute("x").as_float();
52 rigidbody.acceleration.y = node.attribute("y").as_float();
53 }
54 if (const auto node = rigidbodyNode.child("Force")) {
55 rigidbody.force.x = node.attribute("x").as_float();
56 rigidbody.force.y = node.attribute("y").as_float();
57 }
58 if (const auto node = rigidbodyNode.child("AngularVelocity")) {
59 rigidbody.angularVelocity = StringToF32(node.child_value());
60 }
61 if (const auto node = rigidbodyNode.child("AngularAcceleration")) {
62 rigidbody.angularAcceleration = StringToF32(node.child_value());
63 }
64 if (const auto node = rigidbodyNode.child("Torque")) { rigidbody.torque = StringToF32(node.child_value()); }
65 if (const auto node = rigidbodyNode.child("Mass")) { rigidbody.mass = StringToF32(node.child_value()); }
66 if (const auto node = rigidbodyNode.child("InverseMass")) {
67 rigidbody.inverseMass = StringToF32(node.child_value());
68 }
69 if (const auto node = rigidbodyNode.child("Inertia")) { rigidbody.inertia = StringToF32(node.child_value()); }
70 if (const auto node = rigidbodyNode.child("InverseInertia")) {
71 rigidbody.inverseInertia = StringToF32(node.child_value());
72 }
73 if (const auto node = rigidbodyNode.child("Restitution")) {
74 rigidbody.restitution = StringToF32(node.child_value());
75 }
76 if (const auto node = rigidbodyNode.child("Friction")) { rigidbody.friction = StringToF32(node.child_value()); }
77 if (const auto node = rigidbodyNode.child("LinearDamping")) {
78 rigidbody.linearDamping = StringToF32(node.child_value());
79 }
80 if (const auto node = rigidbodyNode.child("AngularDamping")) {
81 rigidbody.angularDamping = StringToF32(node.child_value());
82 }
83 if (const auto node = rigidbodyNode.child("GravityScale")) {
84 rigidbody.gravityScale = StringToF32(node.child_value());
85 }
86 if (const auto node = rigidbodyNode.child("LockRotation")) {

133 static void ParseEntity(const pugi::xml_node& entityNode, EntityDescriptor& entity) {
134 entity.id = entityNode.attribute("id").as_int();
135 entity.name = entityNode.attribute("name").as_string();
136
137 const auto componentsNode = entityNode.child("Components");
138 if (!componentsNode) { throw std::runtime_error("Entity is missing Transform component."); }
139
140 // Parse Transform component
141 if (const auto node = componentsNode.child("Transform")) { entity.transform = ParseTransformComponent(node); }
142
143 // Parse SpriteRenderer component
144 if (const auto node = componentsNode.child("SpriteRenderer")) {
145 entity.spriteRenderer = ParseSpriteRendererComponent(node);
146 }
147
148 if (const auto node = componentsNode.child("Rigidbody2D")) {
149 entity.rigidbody2D = ParseRigidbodyComponent(node);
150 }
151
152 if (const auto node = componentsNode.child("Behavior")) { entity.behavior = ParseBehaviorComponent(node); }
153 }

159 void SceneParser::DescriptorToState(const SceneDescriptor& descriptor,
160 SceneState& outState,
161 ScriptEngine& scriptEngine) {
162 for (const auto& entity : descriptor.entities) {
163 const auto newEntity = outState.CreateEntity(entity.name);
164 auto& [position, rotation, scale] = outState.GetTransform(newEntity);
165 position = entity.transform.position;
166 rotation = entity.transform.rotation;
167 scale = entity.transform.scale;
168
169 if (entity.spriteRenderer.has_value()) {
170 auto& [textureId, geometry] = outState.AddComponent<SpriteRenderer>(newEntity);
171 geometry = Geometry::CreateQuad();
172 textureId = TextureManager::Load(Content::Get<ContentType::Sprite>(entity.spriteRenderer->texture));
173 }
174
175 if (entity.behavior.has_value()) {
176 // Load script
177 const auto scriptPath = Content::Get<ContentType::Script>(entity.behavior->script);
178 const auto scriptSource = IO::ReadString(scriptPath);
179 scriptEngine.LoadScript(scriptSource, entity.behavior->id);
180
181 auto& [id, script] = outState.AddComponent<Behavior>(newEntity);
182 id = entity.behavior->id;
183 script = entity.behavior->script;
184 }
185
186 if (entity.rigidbody2D.has_value()) {
187 auto desc = *entity.rigidbody2D;
188 // Load rigidbody
189 auto& rigidbody = outState.AddComponent<Rigidbody2D>(newEntity);
190 if (desc.type == "Static") {
191 rigidbody.type = BodyType::Static;
192 } else if (desc.type == "Dynamic") {
193 rigidbody.type = BodyType::Dynamic;
194 } else if (desc.type == "Kinematic") {
195 rigidbody.type = BodyType::Kinematic;
196 } else {
197 Log::Error("SceneParser", "BodyType incorrect for Rigidbody2D component: {}", desc.type);
198 return;
199 }
200
201 rigidbody.velocity = desc.velocity;
202 rigidbody.acceleration = desc.acceleration;
203 rigidbody.force = desc.force;
204 rigidbody.angularVelocity = desc.angularVelocity;
205 rigidbody.angularAcceleration = desc.angularAcceleration;
206 rigidbody.torque = desc.torque;
207 rigidbody.mass = desc.mass;
208 rigidbody.inverseMass = desc.inverseMass;
209 rigidbody.inertia = desc.inertia;
210 rigidbody.inverseInertia = desc.inverseInertia;
211 rigidbody.restitution = desc.restitution;
212 rigidbody.friction = desc.friction;
213 rigidbody.linearDamping = desc.linearDamping;
214 rigidbody.angularDamping = desc.angularDamping;
215 rigidbody.gravityScale = desc.gravityScale;
216 rigidbody.lockRotation = desc.lockRotation;
217 }
218 }

225 void SceneParser::DeserializeDescriptor(const fs::path& filename, SceneDescriptor& outDescriptor) {
226 pugi::xml_document doc;
227 const pugi::xml_parse_result result = doc.load_file(filename.string().c_str());
228
229 if (!result) { throw std::runtime_error(fmt::format("XML parsing error: {}", result.description())); }
230
231 const auto sceneNode = doc.child("Scene");
232 if (!sceneNode) { throw std::runtime_error("No scene node found"); }
233
234 // Parse scene name
235 outDescriptor.name = sceneNode.attribute("name").as_string();
236
237 // Parse entities
238 if (const auto entitiesNode = sceneNode.child("Entities")) {
239 for (auto entityNode : entitiesNode.children("Entity")) {
240 EntityDescriptor entity {};
241 entity.id = entityNode.attribute("id").as_int();
242 ParseEntity(entityNode, entity);
243 outDescriptor.entities.push_back(entity);
244 }
245 }
246 }
