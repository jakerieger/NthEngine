143 static void ParseEntityXML(const pugi::xml_node& entityNode, EntityDescriptor& entity) {
144 entity.id = entityNode.attribute("id").as_int();
145 entity.name = entityNode.attribute("name").as_string();
146
147 const auto componentsNode = entityNode.child("Components");
148 if (!componentsNode) {
149 throw std::runtime_error("Entity is missing Transform component.");
150 }
151
152 // Parse Transform component
153 if (const auto node = componentsNode.child("Transform")) {
154 entity.transform = ParseTransformComponentXML(node);
155 }
156
157 // Parse SpriteRenderer component
158 if (const auto node = componentsNode.child("SpriteRenderer")) {
159 entity.spriteRenderer = ParseSpriteRendererComponentXML(node);
160 }
161
162 if (const auto node = componentsNode.child("Rigidbody2D")) {
163 entity.rigidbody2D = ParseRigidbodyComponentXML(node);
164 }
165
166 if (const auto node = componentsNode.child("Behavior")) {
167 entity.behavior = ParseBehaviorComponentXML(node);
168 }

175 void SceneParser::DescriptorToScene(const SceneDescriptor& descriptor, Scene* scene, ScriptEngine& scriptEngine) {
176 for (const auto& entity : descriptor.entities) {
177 auto builder = EntityBuilder::Create(scene, entity.name);
178 builder.SetTransform(entity.transform);
179
180 if (entity.spriteRenderer.has_value()) {
181 builder.AddSpriteRenderer(*entity.spriteRenderer);
182 }
183
184 if (entity.behavior.has_value()) {
185 builder.AddBehavior(*entity.behavior, scriptEngine);
186 }
187
188 if (entity.rigidbody2D.has_value()) {
189 builder.AddRigidbody2D(*entity.rigidbody2D);
190 }
191
192 const auto newEntity = builder.Build();
193 Log::Info("SceneParser", "Loaded entity '{} (ID: {})' to scene state", entity.name, (u32)newEntity);
194 }
195 }

201 void SceneParser::WriteEntityBytes(BinaryWriter& writer, const EntityDescriptor& entity) {
202 writer.WriteUInt32(entity.id);
203 writer.WriteString(entity.name);
204
205 u32 componentCount = 1; // transform is always present
206 const auto componentCountPos = writer.Tell();
207 writer.WriteUInt32(componentCount); // write initial count
208
209 if (entity.spriteRenderer.has_value()) {
210 componentCount++;
211 writer.WriteUInt32(u32(ComponentType::SpriteRenderer));
212 writer.WriteUInt64(entity.spriteRenderer->texture);
213 }
214 if (entity.rigidbody2D.has_value()) {
215 componentCount++;
216 writer.WriteUInt32(u32(ComponentType::Rigidbody2D));
217 writer.WriteUInt32(8);
218 }
219 if (entity.behavior.has_value()) {
220 componentCount++;
221 writer.WriteUInt32(u32(ComponentType::Behavior));
222 writer.WriteUInt64(entity.behavior->script);
223 }
224
225 // update count
226 writer.UpdateAt(componentCountPos, componentCount);
227 }

229 // [Header]
230 // - Magic Number (4 bytes): "SCNE" - for file validation
231 // - Version (4 bytes): Format version for compatibility
232 // - Scene Name Length (4 bytes)
233 // - Scene Name (variable)
234 //
235 // [Entity Count] (4 bytes)
236 //
237 // [For each Entity]
238 // - Entity ID (4 bytes)
239 // - Name Length (4 bytes)
240 // - Name (variable)
241 // - Component Count (4 bytes)
242 //
243 // [For each Component]
244 // - Component Type ID (4 bytes) - enum/hash identifying component type
245 // - Component Data Size (4 bytes) - size of following data block
246 // - Component Data (variable)
247 void SceneParser::SerializeDescriptorBytes(const SceneDescriptor& descriptor, const Path& filename) {
248 const Header header {.sceneNameLength = CAST<u32>(descriptor.name.length()),
249 .sceneName = descriptor.name.c_str()};
250
251 BinaryWriter writer(2_KB);
252 writer.WriteBytes(header.magic, sizeof(header.magic));
253 writer.WriteUInt32(header.version);
254 writer.WriteString(
255 descriptor.name); // WriteString writes the size prefix for us so we can omit writing it here
256 writer.WriteUInt32(descriptor.entities.size());
257
258 for (const auto& entity : descriptor.entities) {
259 WriteEntityBytes(writer, entity);
260 }
261
262 if (!writer.SaveToFile(filename)) {
263 throw std::runtime_error("Failed to save scene descriptor to file");
264 }
265 }

267 void SceneParser::DeserializeDescriptorXML(const Path& filename, SceneDescriptor& outDescriptor) {
268 pugi::xml_document doc;
269 const pugi::xml_parse_result result = doc.load_file(filename.string().c_str());
270
271 if (!result) {
272 throw std::runtime_error(fmt::format("XML parsing error: {}", result.description()));
273 }
274
275 const auto sceneNode = doc.child("Scene");
276 if (!sceneNode) {
277 throw std::runtime_error("No scene node found");
278 }
279
280 // Parse scene name
281 outDescriptor.name = sceneNode.attribute("name").as_string();
282
283 // Parse entities
284 if (const auto entitiesNode = sceneNode.child("Entities")) {
285 for (auto entityNode : entitiesNode.children("Entity")) {
286 EntityDescriptor entity {};
287 entity.id = entityNode.attribute("id").as_int();
288 ParseEntityXML(entityNode, entity);
289 outDescriptor.entities.push_back(entity);
290 }
291 }
292 }
