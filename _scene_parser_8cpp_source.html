148 static void ParseEntityXML(const pugi::xml_node& entityNode, EntityDescriptor& entity) {
149 entity.id = entityNode.attribute("id").as_int();
150 entity.name = entityNode.attribute("name").as_string();
151
152 const auto componentsNode = entityNode.child("Components");
153 if (!componentsNode) {
154 throw std::runtime_error("Entity is missing Transform component.");
155 }
156
157 // Parse Transform component
158 if (const auto node = componentsNode.child("Transform")) {
159 entity.transform = ParseTransformComponentXML(node);
160 }
161
162 // Parse SpriteRenderer component
163 if (const auto node = componentsNode.child("SpriteRenderer")) {
164 entity.spriteRenderer = ParseSpriteRendererComponentXML(node);
165 }
166
167 if (const auto node = componentsNode.child("Rigidbody2D")) {
168 entity.rigidbody2D = ParseRigidbodyComponentXML(node);
169 }
170
171 if (const auto node = componentsNode.child("Behavior")) {
172 entity.behavior = ParseBehaviorComponentXML(node);
173 }

180 void SceneParser::DescriptorToState(const SceneDescriptor& descriptor,
181 SceneState& outState,
182 ScriptEngine& scriptEngine) {
183 for (const auto& entity : descriptor.entities) {
184 const auto newEntity = outState.CreateEntity(entity.name);
185 auto& [position, rotation, scale] = outState.GetTransform(newEntity);
186 position = entity.transform.position;
187 rotation = entity.transform.rotation;
188 scale = entity.transform.scale;
189
190 if (entity.spriteRenderer.has_value()) {
191 auto& [textureId, geometry] = outState.AddComponent<SpriteRenderer>(newEntity);
192 geometry = Geometry::CreateQuad();
193 textureId = TextureManager::Load(Content::Get<ContentType::Sprite>(entity.spriteRenderer->texture));
194 }
195
196 if (entity.behavior.has_value()) {
197 // Load script
198 const auto scriptPath = Content::Get<ContentType::Script>(entity.behavior->script);
199 const auto readScriptResult = IO::ReadText(scriptPath);
200 if (!readScriptResult.has_value()) {
201 Log::Error("SceneParser", "Failed to read script file: {}", readScriptResult.error());
202 }
203 scriptEngine.LoadScript(*readScriptResult, entity.behavior->id);
204
205 auto& [id, script] = outState.AddComponent<Behavior>(newEntity);
206 id = entity.behavior->id;
207 script = entity.behavior->script;
208 }
209
210 if (entity.rigidbody2D.has_value()) {
211 auto desc = *entity.rigidbody2D;
212 // Load rigidbody
213 auto& rigidbody = outState.AddComponent<Rigidbody2D>(newEntity);
214 if (desc.type == "Static") {
215 rigidbody.type = BodyType::Static;
216 } else if (desc.type == "Dynamic") {
217 rigidbody.type = BodyType::Dynamic;
218 } else if (desc.type == "Kinematic") {
219 rigidbody.type = BodyType::Kinematic;
220 } else {
221 Log::Error("SceneParser", "BodyType incorrect for Rigidbody2D component: {}", desc.type);
222 return;
223 }
224
225 rigidbody.velocity = desc.velocity;
226 rigidbody.acceleration = desc.acceleration;
227 rigidbody.force = desc.force;
228 rigidbody.angularVelocity = desc.angularVelocity;
229 rigidbody.angularAcceleration = desc.angularAcceleration;
230 rigidbody.torque = desc.torque;
231 rigidbody.mass = desc.mass;
232 rigidbody.inverseMass = desc.inverseMass;
233 rigidbody.inertia = desc.inertia;
234 rigidbody.inverseInertia = desc.inverseInertia;
235 rigidbody.restitution = desc.restitution;
236 rigidbody.friction = desc.friction;
237 rigidbody.linearDamping = desc.linearDamping;
238 rigidbody.angularDamping = desc.angularDamping;
239 rigidbody.gravityScale = desc.gravityScale;
240 rigidbody.lockRotation = desc.lockRotation;
241 }
242 }

249 void SceneParser::WriteEntityBytes(BinaryWriter& writer, const EntityDescriptor& entity) {
250 writer.WriteUInt32(entity.id);
251 writer.WriteString(entity.name);
252
253 u32 componentCount = 1; // transform is always present
254 const auto componentCountPos = writer.Tell();
255 writer.WriteUInt32(componentCount); // write initial count
256
257 if (entity.spriteRenderer.has_value()) {
258 componentCount++;
259 writer.WriteUInt32(u32(ComponentType::SpriteRenderer));
260 writer.WriteString(entity.spriteRenderer->texture);
261 }
262 if (entity.rigidbody2D.has_value()) {
263 componentCount++;
264 writer.WriteUInt32(u32(ComponentType::Rigidbody2D));
265 writer.WriteUInt32(8);
266 }
267 if (entity.behavior.has_value()) {
268 componentCount++;
269 writer.WriteUInt32(u32(ComponentType::Behavior));
270 writer.WriteUInt32(8 + CAST<u32>(entity.behavior->script.size()));
271 writer.WriteUInt32(entity.behavior->id);
272 writer.WriteString(entity.behavior->script);
273 }
274
275 // update count
276 writer.UpdateAt(componentCountPos, componentCount);
277 }

279 // [Header]
280 // - Magic Number (4 bytes): "SCNE" - for file validation
281 // - Version (4 bytes): Format version for compatibility
282 // - Scene Name Length (4 bytes)
283 // - Scene Name (variable)
284 //
285 // [Entity Count] (4 bytes)
286 //
287 // [For each Entity]
288 // - Entity ID (4 bytes)
289 // - Name Length (4 bytes)
290 // - Name (variable)
291 // - Component Count (4 bytes)
292 //
293 // [For each Component]
294 // - Component Type ID (4 bytes) - enum/hash identifying component type
295 // - Component Data Size (4 bytes) - size of following data block
296 // - Component Data (variable)
297 void SceneParser::SerializeDescriptorBytes(const SceneDescriptor& descriptor, const Path& filename) {
298 const Header header {.sceneNameLength = CAST<u32>(descriptor.name.length()),
299 .sceneName = descriptor.name.c_str()};
300
301 BinaryWriter writer(2_KB);
302 writer.WriteBytes(header.magic, sizeof(header.magic));
303 writer.WriteUInt32(header.version);
304 writer.WriteString(
305 descriptor.name); // WriteString writes the size prefix for us so we can omit writing it here
306 writer.WriteUInt32(descriptor.entities.size());
307
308 for (const auto& entity : descriptor.entities) {
309 WriteEntityBytes(writer, entity);
310 }
311
312 if (!writer.SaveToFile(filename)) {
313 throw std::runtime_error("Failed to save scene descriptor to file");
314 }
315 }

317 void SceneParser::DeserializeDescriptorXML(const Path& filename, SceneDescriptor& outDescriptor) {
318 pugi::xml_document doc;
319 const pugi::xml_parse_result result = doc.load_file(filename.string().c_str());
320
321 if (!result) {
322 throw std::runtime_error(fmt::format("XML parsing error: {}", result.description()));
323 }
324
325 const auto sceneNode = doc.child("Scene");
326 if (!sceneNode) {
327 throw std::runtime_error("No scene node found");
328 }
329
330 // Parse scene name
331 outDescriptor.name = sceneNode.attribute("name").as_string();
332
333 // Parse entities
334 if (const auto entitiesNode = sceneNode.child("Entities")) {
335 for (auto entityNode : entitiesNode.children("Entity")) {
336 EntityDescriptor entity {};
337 entity.id = entityNode.attribute("id").as_int();
338 ParseEntityXML(entityNode, entity);
339 outDescriptor.entities.push_back(entity);
340 }
341 }
342 }
