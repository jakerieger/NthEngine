78 void Game::LoadPlugins() {
79 const auto pluginsPath = fs::current_path() / "Plugins";
80 if (!exists(pluginsPath)) {
81 Log::Warn("Game", "No plugins directory found");
82 } else {
83 for (const auto& entry : fs::directory_iterator(pluginsPath)) {
84 if (entry.is_regular_file() && entry.path().extension() == ".plugin") {
85 Plugin plugin;
86 const auto loadResult = plugin.Load(entry.path().string());
87 if (!loadResult.has_value()) {
88 Log::Error("Game", "Error loading plugin: {}", loadResult.error());
89 continue;
90 }
91
92 Log::Debug("Game", "Loaded plugin: '{}' (from: {})", *loadResult, plugin.GetPluginPath().string());
93
94 mPlugins[*loadResult] = std::move(plugin);
95 }
96 }
97 }
98 }

100 void Game::UpdateSceneCache() {
101 mSceneCache.clear();
102 bool loadedEntry = false;
103 for (auto& scenePath : AssetManager::GetScenes()) {
104 SceneDescriptor desc;
105 SceneParser::DeserializeDescriptorXML(scenePath, desc);
106 mSceneCache[desc.name] = desc;
107 if (desc.entry && !loadedEntry) {
108 mActiveScene->LoadDescriptor(desc, GetScriptEngine());
109 loadedEntry = true;
110 }
111 Log::Debug("Game", "Added scene '{}' to cache", desc.name);
112 }
113 }

128 void Game::OnAwake() {
129 Log::Info("Game", "Initializing game systems...");
130
131 // Get window dimensions from Window base class
132 u32 width, height;
133 GetSize(width, height);
134
135 // Create main render target
136 mMainRenderTarget = make_unique<RenderTarget>(RenderTargetConfig {.type = RenderTargetType::Window,
137 .width = width,
138 .height = height,
139 .enableDepth = true,
140 .enableStencil = false});
141
142 if (!mMainRenderTarget->Initialize()) {
143 Log::Critical("Game", "Failed to initialize main render target");
144 Close();
145 return;
146 }
147
148 // Initialize asset managers
149 TextureManager::Initialize();
150 ShaderManager::Initialize();
151
152 // Initialize audio engine
153 mAudioEngine.Initialize();
154
155 // Initialize script engine
156 InitializeScriptEngine();
157
158 LoadEngineConfigurations();
159
160 // Asset manager
161 if (!AssetManager::Initialize()) {
162 Log::Critical("Game", "Failed to initialize asset manager");
163 Close();
164 return;
165 }
166
167 // Debug layers
168 LoadDebugLayers(width, height);
169
170 // Create the initial scene
171 mActiveScene = make_unique<Scene>(GetRenderContext());
172
173 UpdateSceneCache();
174
175 // Initialize job system
176 gJobSystem = make_unique<JobSystem>();
177 gJobSystem->Initialize();
178
179 LoadPlugins();
180
181 Log::Debug("Game",
182 "Successfully initialized game instance:\n-- Dimensions: {}x{}\n-- V-Sync: {}\n-- Worker Threads: "
183 "{}\n-- Plugins: {}",
184 width,
185 height,
186 mVsync ? "On" : "Off",
187 gJobSystem->GetWorkerCount(),
188 mPlugins.size());
189
190 for (const auto& plugin : mPlugins | std::views::values) {
191 plugin->OnEngineStart(this);
192 }
193
194 // Call user's OnAwake if they have overridden it
195 if (mActiveScene)
196 mActiveScene->Awake(GetScriptEngine());
197
198 for (const auto& plugin : mPlugins | std::views::values) {
199 plugin->OnSceneAwake(this);
200 }
201
202 LoadContent();
203 }

205 void Game::OnUpdate(const Clock& clock) {
206 // update debug ui
207 mImGuiDebugLayer->UpdateFrameRate((f32)clock.GetFramesPerSecond());
208 const auto fT = (1.f / clock.GetFramesPerSecond()) * 1000.f;
209 mImGuiDebugLayer->UpdateFrameTime((f32)fT);
210 // silly hack, will calculate once threading is actually implemented
211 mImGuiDebugLayer->UpdateMainThreadTime((f32)fT / 2.f);
212 mImGuiDebugLayer->UpdateRenderThreadTime((f32)fT / 2.f);
213 mImGuiDebugLayer->UpdateEntities(mActiveScene->GetState().GetEntityCount());
214
215 auto& resMgr = mActiveScene->GetResourceManager();
216 mImGuiDebugLayer->UpdateResourcePoolAllocatedBytes(resMgr.GetAllocator().GetSize());
217 mImGuiDebugLayer->UpdateResourcePoolUsedBytes(resMgr.GetAllocator().GetUsedMemory());
218
219 if (mActiveScene) {
220 mActiveScene->Update(clock, GetScriptEngine());
221
222 vector<Transform> transforms;
223 const auto iter = mActiveScene->GetState().View<Transform>().each();
224 for (auto [entity, transform] : iter) {
225 transforms.push_back(transform);
226 }
227 mPhysicsDebugLayer->UpdateTransforms(transforms);
228 }
229
230 for (const auto& plugin : mPlugins | std::views::values) {
231 plugin->OnSceneUpdate(this);
232 }
233
234 // Advance frame allocator
235 mFrameAllocator.NextFrame();
236 }

300 mPhysicsDebugLayer->OnRender();
301
302 glfwSwapBuffers(GetHandle());
303
304 CommandExecutor::gDrawCalls = 0;

307 bool Game::InitializeScriptEngine() {
308 mScriptEngine.Initialize();
309 if (!mScriptEngine.IsInitialized())
310 return false;
311
312 // Register globals
313 auto& lua = mScriptEngine.GetLuaState();
314 auto gameGlobal = lua.new_usertype<Game>("Game");
315 gameGlobal["Quit"] = [this] { Quit(); };
316 gameGlobal["GetScreenSize"] = [this]() -> Vec2 {
317 u32 w, h;
318 GetSize(w, h);
319 return {CAST<f32>(w), CAST<f32>(h)};
320 };
321 gameGlobal["LoadScene"] = [this](const sol::object&, const string& sceneName) -> bool {
322 return LoadScene(sceneName);
323 };
324
325 lua["Scene"] = &mActiveScene->GetState();
326
327 // Register globals
328 Log::RegisterLuaGlobals(lua);
329 Math::RegisterLuaGlobals(lua);
330 Coordinates::RegisterLuaGlobals(lua);
331 GetInputManager().RegisterLuaGlobals(lua); // Use Window's InputManager
332 mAudioEngine.RegisterLuaGlobals(lua);
333 ScriptTypeRegistry::RegisterTypes(mScriptEngine);
334
335 return true;
336 }
