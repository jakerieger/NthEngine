46 void Game::Run() {
47 mRunning = Initialize();
48 if (!mRunning)
49 return;
50 OnAwake();
51 {
52 // Main loop
53 // ReSharper disable once CppDFAConstantConditions
54 while (mRunning && !glfwWindowShouldClose(mWindow)) {
55 mClock.Tick();
56 OnUpdate(mClock);
57
58 Render();
59
60 OnLateUpdate();
61 glfwPollEvents();
62
63 mFrameAllocator.NextFrame();
64 }
65
66 mRunning = false;
67 }
68 OnDestroyed();
69 Shutdown();
70 }

137 void Game::OnUpdate(const Clock& clock) {
138 mDebugManager.Update(clock.GetDeltaTime());
139
140 if (mActiveScene) {
141 mActiveScene->Update(clock, GetScriptEngine());
142
143 vector<Transform> transforms;
144 const auto iter = mActiveScene->GetState().View<Transform>().each();
145 for (auto [entity, transform] : iter) {
146 transforms.push_back(transform);
147 }
148 mPhysicsDebugLayer->UpdateTransforms(transforms);
149 }

162 bool Game::Initialize() {
163 if (!glfwInit()) {
164 Log::Critical("Game", "Failed to initialize GLFW");
165 return false;
166 }
167
168 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
169 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
170 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
171#ifdef __APPLE__
172 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
173#endif
174
175 mWindow = glfwCreateWindow((i32)mWidth, (i32)mHeight, mTitle.c_str(), nullptr, nullptr);
176 if (!mWindow) {
177 glfwTerminate();
178 Log::Critical("Game", "Failed to create GLFW mWindow");
179 return false;
180 }
181
182 glfwMakeContextCurrent(mWindow);
183 glfwSetWindowUserPointer(mWindow, this);
184
185 glfwSetFramebufferSizeCallback(mWindow, GLFWResizeCallback);
186 glfwSetKeyCallback(mWindow, GLFWKeyCallback);
187 glfwSetMouseButtonCallback(mWindow, GLFWMouseButtonCallback);
188 glfwSetCursorPosCallback(mWindow, GLFWMouseCursorCallback);
189 glfwSetScrollCallback(mWindow, GLFWMouseScrollCallback);
190
191 if (mVsync)
192 glfwSwapInterval(1);
193 else
194 glfwSwapInterval(0);
195
196 if (!mRenderContext.Initialize(mWidth, mHeight)) {
197 glfwDestroyWindow(mWindow);
198 glfwTerminate();
199 Log::Critical("Game", "Failed to initialize render context");
200 return false;
201 }
202
203 TextureManager::Initialize();
204 ShaderManager::Initialize();
205 mAudioEngine.Initialize();
206 InitializeScriptEngine();
207
208 // Asset manager
209 if (!AssetManager::Initialize()) {
210 Log::Critical("Game", "Failed to initialize asset manager");
211 glfwDestroyWindow(mWindow);
212 glfwTerminate();
213 mRenderContext.Shutdown();
214 TextureManager::Shutdown();
215 ShaderManager::Shutdown();
216 mAudioEngine.Shutdown();
217 return false;
218 }
219
220 // Debug layers
221 mImGuiDebugLayer = make_unique<ImGuiDebugLayer>(GetWindowHandle());
222 mDebugManager.AttachOverlay("ImGuiDebugLayer", mImGuiDebugLayer.get());
223
224 mPhysicsDebugLayer = make_unique<PhysicsDebugLayer>(mWidth, mHeight);
225 mDebugManager.AttachOverlay("PhysicsDebugLayer", mPhysicsDebugLayer.get());
226
227 // Create the initial scene
228 mActiveScene = make_unique<Scene>(GetRenderContext());
229
230 gJobSystem = make_unique<JobSystem>();
231 gJobSystem->Initialize();
232
233 Log::Debug(
234 "Game",
235 "Successfully initialized game instance:\n-- Dimensions: {}x{}\n-- V-Sync: {}\n-- Worker Threads: {}",
236 mWidth,
237 mHeight,
238 mVsync ? "On" : "Off",
239 gJobSystem->GetWorkerCount());
240
241 return true;
242 }

257 }
258
259 if (mWindow)
260 glfwDestroyWindow(mWindow);
261 glfwTerminate();
262
263 Log::Shutdown();
264 }

266 bool Game::InitializeScriptEngine() {
267 mScriptEngine.Initialize();
268 if (!mScriptEngine.IsInitialized())
269 return false;
270
271 // Register globals
272 auto& lua = mScriptEngine.GetLuaState();
273 auto gameGlobal = lua.new_usertype<Game>("Game");
274 gameGlobal["Quit"] = [this] { Quit(); };
275 gameGlobal["GetScreenSize"] = [this]() -> Vec2 { return {mWidth, mHeight}; };
276
277 lua["Scene"] = &mActiveScene->GetState();
278
279 // Register globals
280 Log::RegisterLuaGlobals(lua);
281 Math::RegisterLuaGlobals(lua);
282 Coordinates::RegisterLuaGlobals(lua);
283 mInputManager.RegisterLuaGlobals(lua);
284 mAudioEngine.RegisterLuaGlobals(lua);
285 ScriptTypeRegistry::RegisterTypes(mScriptEngine);
286
287 return true;
288 }

304 void Game::SetWindowIcon(const fs::path& filename) const {
305 i32 width, height, channels;
306 u8* pixels = stbi_load(filename.string().c_str(), &width, &height, &channels, 4); // Force RGBA
307
308 if (pixels) {
309 GLFWimage icon;
310 icon.width = width;
311 icon.height = height;
312 icon.pixels = pixels;
313 glfwSetWindowIcon(mWindow, 1, &icon);
314 stbi_image_free(pixels);
315 } else {
316 Log::Error("Game", "Failed to load window icon: {}", filename.string());
317 }
318 }
