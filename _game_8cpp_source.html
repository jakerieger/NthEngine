44 void Game::Run() {
45 mRunning = Initialize();
46 if (!mRunning) return;
47 OnAwake();
48 {
49 // Main loop
50 // ReSharper disable once CppDFAConstantConditions
51 while (mRunning && !glfwWindowShouldClose(mWindow)) {
52 mClock.Tick();
53 OnUpdate(mClock);
54
55 Render();
56
57 OnLateUpdate();
58 glfwPollEvents();
59 }
60
61 mRunning = false;
62 }
63 OnDestroyed();
64 Shutdown();
65 }

131 void Game::OnUpdate(const Clock& clock) {
132 mDebugManager.Update(clock.GetDeltaTime());
133
134 if (mActiveScene) {
135 mActiveScene->Update(mScriptEngine, clock);
136
137 vector<Transform> transforms;
138 const auto iter = mActiveScene->GetState().View<Transform>().each();
139 for (auto [entity, transform] : iter) {
140 transforms.push_back(transform);
141 }
142 mPhysicsDebugLayer->UpdateTransforms(transforms);
143 }

154 bool Game::Initialize() {
155 // Enforce X11 on Wayland systems for debugging support (RenderDoc doens't support Wayland)
156#if defined(N_ENGINE_PLATFORM_LINUX_WAYLAND) || defined(N_ENGINE_PLATFORM_LINUX_X11)
157 glfwInitHint(GLFW_PLATFORM, GLFW_PLATFORM_X11);
158#endif
159
160 if (!glfwInit()) {
161 Log::Critical("Game", "Failed to initialize GLFW");
162 return false;
163 }
164
165 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
166 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
167 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
168#ifdef __APPLE__
169 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
170#endif
171
172 mWindow = glfwCreateWindow((i32)mWidth, (i32)mHeight, mTitle.c_str(), nullptr, nullptr);
173 if (!mWindow) {
174 glfwTerminate();
175 Log::Critical("Game", "Failed to create GLFW mWindow");
176 return false;
177 }
178
179 glfwMakeContextCurrent(mWindow);
180 glfwSetWindowUserPointer(mWindow, this);
181
182 glfwSetFramebufferSizeCallback(mWindow, GLFWResizeCallback);
183 glfwSetKeyCallback(mWindow, GLFWKeyCallback);
184 glfwSetMouseButtonCallback(mWindow, GLFWMouseButtonCallback);
185 glfwSetCursorPosCallback(mWindow, GLFWMouseCursorCallback);
186 glfwSetScrollCallback(mWindow, GLFWMouseScrollCallback);
187
188 if (mVsync) glfwSwapInterval(1);
189 else glfwSwapInterval(0);
190
191 if (!mRenderContext.Initialize(mWidth, mHeight)) {
192 glfwDestroyWindow(mWindow);
193 glfwTerminate();
194 Log::Critical("Game", "Failed to initialize render context");
195 return false;
196 }
197
198 TextureManager::Initialize();
199 ShaderManager::Initialize();
200 mAudioEngine.Initialize();
201 InitializeScriptEngine();
202
203 // Debug layers
204 mImGuiDebugLayer = make_unique<ImGuiDebugLayer>(GetWindowHandle());
205 mDebugManager.AttachOverlay("ImGuiDebugLayer", mImGuiDebugLayer.get());
206
207 mPhysicsDebugLayer = make_unique<PhysicsDebugLayer>(mWidth, mHeight);
208 mDebugManager.AttachOverlay("PhysicsDebugLayer", mPhysicsDebugLayer.get());
209
210 Log::Debug("Game",
211 "Successfully initialized game instance:\n-- Dimensions: {}x{}\n-- V-Sync: {}",
212 mWidth,
213 mHeight,
214 mVsync ? "On" : "Off");
215
216 return true;
217 }

234 bool Game::InitializeScriptEngine() {
235 mScriptEngine.Initialize();
236 if (!mScriptEngine.IsInitialized()) return false;
237
238 // Register globals
239 auto& lua = mScriptEngine.GetLuaState();
240 auto gameGlobal = lua.new_usertype<Game>("Game");
241 gameGlobal["Quit"] = [this] { Quit(); };
242 gameGlobal["GetScreenSize"] = [this]() -> Vec2 { return {mWidth, mHeight}; };
243
244 lua["Scene"] = &mActiveScene->GetState();
245
246 // Register globals
247 Log::RegisterLuaGlobals(lua);
248 Math::RegisterLuaGlobals(lua);
249 Coordinates::RegisterLuaGlobals(lua);
250 mInputManager.RegisterLuaGlobals(lua);
251 mAudioEngine.RegisterLuaGlobals(lua);
252 ScriptTypeRegistry::RegisterTypes(mScriptEngine);
253
254 return true;
255 }

269 void Game::SetWindowIcon(const fs::path& filename) const {
270 i32 width, height, channels;
271 u8* pixels = stbi_load(filename.string().c_str(), &width, &height, &channels, 4); // Force RGBA
272
273 if (pixels) {
274 GLFWimage icon;
275 icon.width = width;
276 icon.height = height;
277 icon.pixels = pixels;
278 glfwSetWindowIcon(mWindow, 1, &icon);
279 stbi_image_free(pixels);
280 } else {
281 Log::Error("Game", "Failed to load window icon: {}", filename.string());
282 }
283 }
