34 bool RenderContext::Initialize(u32 width, u32 height) {
35 if (mInitialized) return true;
36
37 mWidth = width;
38 mHeight = height;
39
40 if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress)) {
41 Log::Critical("RenderContext", "Failed to load GLAD");
42 return false;
43 }
44
45 GLCall(glViewport, 0, 0, (i32)mWidth, (i32)mHeight);
46
47 // Enable depth testing (common OpenGL setup)
48 GLCall(glEnable, GL_DEPTH_TEST);
49
50 // Enable blending for transparency
51 GLCall(glEnable, GL_BLEND);
52 GLCall(glBlendFunc, GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
53
54 mCommandQueue.InitializeBatchResources();
55 mCommandQueue.Reserve(1000);
56 mInitialized = true;
57
58 return true;
59 }

66 void RenderContext::BeginFrame() {
