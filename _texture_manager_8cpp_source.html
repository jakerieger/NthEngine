58 TextureID TextureManager::Load(const fs::path& filename) {
59 u32 id;
60 glGenTextures(1, &id);
61
62 int w, h, channels;
63 stbi_set_flip_vertically_on_load(true); // For OpenGL
64 const auto data = stbi_load(filename.string().c_str(), &w, &h, &channels, 0);
65 if (!data) { Log::Error("TextureManager", "Failed to load image: `{}`", filename.string()); }
66
67 GLenum format = GL_RGBA;
68 if (channels == 1) format = GL_RED;
69 else if (channels == 3) format = GL_RGB;
70 else if (channels == 4) format = GL_RGBA;
71
72 GLCall(glBindTexture, GL_TEXTURE_2D, id);
73 GLCall(glTexImage2D, GL_TEXTURE_2D, 0, CAST<int>(format), w, h, 0, format, GL_UNSIGNED_BYTE, data);
74 GLCall(glGenerateMipmap, GL_TEXTURE_2D);
75
76 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, format == GL_RGBA ? GL_CLAMP_TO_EDGE : GL_REPEAT);
77 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, format == GL_RGBA ? GL_CLAMP_TO_EDGE : GL_REPEAT);
78 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
79 GLCall(glTexParameteri, GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
80
81 stbi_image_free(data);
82
83 sCache[filename.string()] = id;
84 Log::Debug("TextureManager", "Loaded texture `{}` with OpenGL id `{}`", filename.string().c_str(), id);
85
86 return id;
87 }
