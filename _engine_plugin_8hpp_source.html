55namespace Astera {
56 template<typename Plugin>
57 concept ValidPlugin = std::is_base_of_v<IEnginePlugin, Plugin>;
58
59 class Plugin {
60 IEnginePlugin* mPluginHandle {nullptr};
61 LibraryHandle mLibHandle {nullptr};
62 Path mPluginPath;
63
64 public:
65 Plugin() = default;
66
67 Result<string> Load(const Path& pluginFile) {
68 mLibHandle = PLUGIN_LOAD(pluginFile.c_str());
69 if (!mLibHandle) {
70 return unexpected(fmt::format("Plugin '{}' failed to load. Ensure the plugin is a valid Astera plugin.",
71 pluginFile.string()));
72 }
73
74 const auto CreatePlugin = PLUGIN_GET_CREATE_FUNC(mLibHandle);
75 if (!CreatePlugin) {
76 PLUGIN_UNLOAD(mLibHandle);
77 return unexpected(
78 "Could not find 'CreatePlugin' symbol in plugin. Make sure it is defined and exported.");
79 }
80
81 mPluginHandle = CreatePlugin();
82 mPluginPath = pluginFile;
83 return mPluginHandle->GetName();
84 }
85
86 i32 Unload() {
87 if (mLibHandle) {
88 PLUGIN_UNLOAD(mLibHandle);
89 mLibHandle = nullptr;
90 }
91
92 if (mPluginHandle) {
93 mPluginHandle = nullptr;
94 }
95
96 mPluginPath = "";
97 return 0;
98 }
99
100 ASTERA_CLASS_PREVENT_COPIES(Plugin)
101
102 ~Plugin() {
103 std::ignore = Unload();
104 }
105
106 Plugin(Plugin&& other) noexcept
107 : mPluginHandle(std::exchange(other.mPluginHandle, nullptr)),
108 mLibHandle(std::exchange(other.mLibHandle, nullptr)) {}
109
110 Plugin& operator=(Plugin&& other) noexcept {
111 if (this != &other) {
112 std::ignore = Unload();
113 mPluginHandle = std::exchange(other.mPluginHandle, nullptr);
114 mLibHandle = std::exchange(other.mLibHandle, nullptr);
115 }
116 return *this;
117 }
118
