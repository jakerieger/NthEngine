51 void CommandQueue::ExecuteQueueBatched() {
52 if (!mBatchResourcesInitialized) {
53 Log::Error("CommandQueue", "Batch resources not initialized");
54 ExecuteQueue(); // Fallback to non-batched
55 return;
56 }
57
58 CommandExecutor executor;
59
60 // Execute non-sprite commands first
61 for (const auto& command : mCommands) {
62 if (!std::holds_alternative<DrawSpriteCommand>(command)) {
63 std::visit(executor, command);
64 }

66
67 // Batch and render sprites
68 BatchSpriteCommands();
69
70 for (const auto& batch : mBatches) {
71 RenderBatch(batch);
72 }
73
74 Clear();
75 }

86 void CommandQueue::BatchSpriteCommands() {
87 mBatches.clear();
88
89 // Extract sprite commands and sort by texture
90 vector<size_t> spriteIndices;
91 for (size_t i = 0; i < mCommands.size(); ++i) {
92 if (std::holds_alternative<DrawSpriteCommand>(mCommands[i])) {
93 spriteIndices.push_back(i);
94 }
95 }
96
97 if (spriteIndices.empty())
98 return;
99
100 // Stable sort to maintain draw order for same texture
101 std::ranges::stable_sort(spriteIndices, [this](size_t a, size_t b) {
102 const auto& cmdA = std::get<DrawSpriteCommand>(mCommands[a]);
103 const auto& cmdB = std::get<DrawSpriteCommand>(mCommands[b]);
104 return cmdA.spriteRenderer->sprite->GetID() < cmdB.spriteRenderer->sprite->GetID();
105 });
106
107 // Build batches
108 SpriteBatch currentBatch;
109 currentBatch.quadVAO = mBatchVAO;
110 u32 currentTexture = static_cast<u32>(-1);
111
112 for (const size_t idx : spriteIndices) {
113 const auto& cmd = std::get<DrawSpriteCommand>(mCommands[idx]);
114
115 // Start new batch if texture changes or batch is full
116 if (cmd.spriteRenderer->sprite->GetID() != currentTexture ||
117 currentBatch.SpriteCount() >= kMaxSpritesPerBatch) {
118 if (!currentBatch.instances.empty()) {
119 mBatches.push_back(std::move(currentBatch));
120 currentBatch = SpriteBatch();
121 currentBatch.quadVAO = mBatchVAO;
122 }
123
124 currentTexture = cmd.spriteRenderer->sprite->GetID();
125 currentBatch.textureId = currentTexture;
126 }
127
128 // Calculate MVP transform for this sprite
129 const Mat4 model = cmd.transform->GetMatrix();
130 const Mat4 projection = Coordinates::CreateScreenProjection(cmd.screenDimensions.x, cmd.screenDimensions.y);
131 const Mat4 mvp = projection * model;
132
133 // Add instance data
134 currentBatch.instances.emplace_back(mvp, cmd.tintColor);
135 }
136
137 // Add final batch
138 if (!currentBatch.instances.empty()) {
139 mBatches.push_back(std::move(currentBatch));
140 }
141
142 // Log::Debug("CommandQueue", "Batched {} sprites into {} draw calls", spriteIndices.size(), mBatches.size());
143 }

145 void CommandQueue::RenderBatch(const SpriteBatch& batch) const {
146 if (batch.instances.empty())
147 return;
148
149 ASTERA_ASSERT(mBatchResourcesInitialized);
150 ASTERA_ASSERT(batch.SpriteCount() <= kMaxSpritesPerBatch);
151
152 // Upload instance data
153 mInstanceVBO->Bind();
154 mInstanceVBO->UpdateData(batch.instances.data(), batch.instances.size() * sizeof(SpriteInstanceData), 0);
155
156 // Bind shader
157 const auto shader = ShaderManager::GetShader(Shaders::SpriteInstanced);
158 ASTERA_ASSERT(shader);
159 shader->Bind();
160
161 // Bind texture
162 GLCall(glActiveTexture, GL_TEXTURE0);
163 GLCall(glBindTexture, GL_TEXTURE_2D, batch.textureId);
164 shader->SetUniform("uSprite", 0);
165
166 // Draw instanced
167 batch.quadVAO->Bind();
168 GLCall(glDrawElementsInstanced,
169 GL_TRIANGLES,
170 6, // 6 indices per quad
171 GL_UNSIGNED_INT,
172 nullptr,
173 CAST<GLsizei>(batch.SpriteCount()));
174 CommandExecutor::gDrawCalls++;
175
176 shader->Unbind();
177 }

179 void CommandQueue::InitializeBatchResources() {
180 if (mBatchResourcesInitialized)
181 return;
182
183 Log::Debug("CommandQueue", "Initializing sprite batching resources...");
184
185 // Create shared quad geometry (all sprites share this)
186 const SpriteVertex quadVertices[] = {
187 {-0.5f, -0.5f, 0.0f, 0.0f}, // Bottom-left
188 {0.5f, -0.5f, 1.0f, 0.0f}, // Bottom-right
189 {-0.5f, 0.5f, 0.0f, 1.0f}, // Top-left
190 {0.5f, 0.5f, 1.0f, 1.0f}, // Top-right
191 };
192
193 const u32 quadIndices[] = {
194 0,
195 1,
196 2, // First triangle
197 2,
198 1,
199 3 // Second triangle
200 };
201
202 // Create quad vertex buffer
203 mQuadVBO = make_shared<VertexBuffer>();
204 mQuadVBO->SetData(quadVertices, sizeof(quadVertices), BufferUsage::Static);
205
206 // Create quad index buffer
207 mQuadIBO = make_shared<IndexBuffer>();
208 mQuadIBO->SetIndices(quadIndices, 6, BufferUsage::Static);
209
210 // Create instance data buffer (dynamic, updated per frame)
211 const size_t instanceBufferSize = kMaxSpritesPerBatch * sizeof(SpriteInstanceData);
212 mInstanceVBO = make_shared<VertexBuffer>();
213 mInstanceVBO->SetData(nullptr, instanceBufferSize, BufferUsage::Stream);
214
215 // Setup VAO
216 mBatchVAO = make_shared<VertexArray>();
217
218 // Add quad vertices (per-vertex attributes)
219 VertexLayout quadLayout;
220 quadLayout.AddAttribute(VertexAttribute("aVertex", AttributeType::Float4));
221 mBatchVAO->AddVertexBuffer(mQuadVBO, quadLayout);
222
223 // Add instance data (per-instance attributes)
224 // Note: We'll set up instanced attributes manually since VertexLayout doesn't support divisors yet
225 mBatchVAO->Bind();
226 mInstanceVBO->Bind();
227
228 // Layout for SpriteInstanceData:
229 // - Mat4 transform (location 1-4, 4 vec4s)
230 // - Vec4 tintColor (location 5)
231
232 const size_t mat4Size = sizeof(Mat4);
233 const size_t vec4Size = sizeof(Vec4);
234 const size_t stride = sizeof(SpriteInstanceData);
235
236 // Setup transform matrix (takes 4 attribute locations)
237 for (u32 i = 0; i < 4; ++i) {
238 GLCall(glEnableVertexAttribArray, 1 + i);
239 GLCall(glVertexAttribPointer,
240 1 + i,
241 4,
242 GL_FLOAT,
243 GL_FALSE,
244 CAST<GLsizei>(stride),
245 RCAST<void*>(i * vec4Size));
246 GLCall(glVertexAttribDivisor, 1 + i, 1); // One per instance
247 }
248
249 // Setup tint color
250 GLCall(glEnableVertexAttribArray, 5);
251 GLCall(glVertexAttribPointer, 5, 4, GL_FLOAT, GL_FALSE, CAST<GLsizei>(stride), RCAST<void*>(mat4Size));
252 GLCall(glVertexAttribDivisor, 5, 1); // One per instance
253
254 // Set index buffer
255 mBatchVAO->SetIndexBuffer(mQuadIBO);
256
257 VertexArray::Unbind();
258
259 mBatchResourcesInitialized = true;
260 Log::Debug("CommandQueue", "Sprite batching initialized (max {} sprites per batch)", kMaxSpritesPerBatch);
261 }

284 }
285

288 }

293 void CommandExecutor::operator()(const DrawSpriteCommand& cmd) const {
294 const auto spriteShader = ShaderManager::GetShader(Shaders::Sprite);
295 ASTERA_ASSERT(spriteShader);
296 spriteShader->Bind();
297
298 cmd.spriteRenderer->sprite->Bind(0);
299 spriteShader->SetUniform("uSprite", 0);
300
301 const Mat4 model = cmd.transform->GetMatrix();
302 const Mat4 projection = Coordinates::CreateScreenProjection(cmd.screenDimensions.x, cmd.screenDimensions.y);
303 const Mat4 mvp = projection * model;
304 spriteShader->SetUniform("uMVP", mvp);
305
306 const auto drawCmd = DrawIndexedCommand {
307 .vao = cmd.spriteRenderer->geometry->GetVertexArray(),
308 .indexCount = CAST<u32>(cmd.spriteRenderer->geometry->GetVertexArray()->GetIndexBuffer()->GetCount()),
309 .primitiveType = GL_TRIANGLES,
310 .indexOffset = 0};
311
312 (*this)(drawCmd);
313 }

323 void CommandExecutor::operator()(const SetUniformCommand& cmd) const {
324 GLCall(glUseProgram, cmd.programId);
325
326 const GLint location = glGetUniformLocation(cmd.programId, cmd.name.c_str());
327 if (location == -1) {
328 Log::Warn("CommandQueue", "Uniform '{}' not found in shader program {}", cmd.name, cmd.programId);
329 return;
330 }
331
332 // Use std::visit to handle the different uniform value types
333 std::visit(
334 [location]<typename T0>(const T0& value) {
335 using T = std::decay_t<T0>;
336
337 if constexpr (std::is_same_v<T, i32>) {
338 GLCall(glUniform1i, location, value);
339 } else if constexpr (std::is_same_v<T, f32>) {
340 GLCall(glUniform1f, location, value);
341 } else if constexpr (std::is_same_v<T, glm::vec2>) {
342 GLCall(glUniform2f, location, value.x, value.y);
343 } else if constexpr (std::is_same_v<T, glm::vec3>) {
344 GLCall(glUniform3f, location, value.x, value.y, value.z);
345 } else if constexpr (std::is_same_v<T, glm::vec4>) {
346 GLCall(glUniform4f, location, value.x, value.y, value.z, value.w);
347 } else if constexpr (std::is_same_v<T, glm::mat4>) {
348 GLCall(glUniformMatrix4fv, location, 1, GL_FALSE, &value[0][0]);
349 }
350 },
351 cmd.value);
352 }
