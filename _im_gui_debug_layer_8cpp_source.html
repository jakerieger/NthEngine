61 void ImGuiDebugLayer::InitImGui(GLFWwindow* window) const {
62 N_UNUSED(this);
63
64 IMGUI_CHECKVERSION();
65 ImGui::CreateContext();
66 ImGuiIO& io = ImGui::GetIO();
67 io.ConfigFlags |= ImGuiConfigFlags_DockingEnable;
68 io.IniFilename = nullptr;
69
70 ImGui_ImplGlfw_InitForOpenGL(window, true);
71 ImGui_ImplOpenGL3_Init("#version 460");
72
73 Log::Info("ImGuiDebugLayer", "Debug layer initialized");
74 }

83 void ImGuiDebugLayer::DrawPerformanceOverlay() {
84 ImGui::SetNextWindowPos({0, 0});
85 ImGui::Begin("Performance",
86 nullptr,
87 ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoDecoration | ImGuiWindowFlags_NoMove |
88 ImGuiWindowFlags_AlwaysAutoResize | ImGuiWindowFlags_NoBackground);
89
90 ImGui::Text("%.1f FPS", ImGui::GetIO().Framerate);
91
92 static float values[90] = {};
93 static int values_offset = 0;
94 values[values_offset] = ImGui::GetIO().Framerate;
95 values_offset = (values_offset + 1) % IM_ARRAYSIZE(values);
96
97 ImGui::PlotLines("FPS", values, IM_ARRAYSIZE(values), values_offset, nullptr, 0.0f, 100.0f, ImVec2(0, 80));
98
99 ImGui::End();
100 }
