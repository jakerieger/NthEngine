87 static f32 CalcBytesOOM(u64 value, string& suffix) {
88 if (value <= 1_KB) {
89 suffix = "B";
90 return f32(value);
91 }
92
93 if (value <= 1_MB) {
94 suffix = "KB";
95 return f32(value) / 1_KB;
96 }
97
98 if (value <= 1_GB) {
99 suffix = "MB";
100 return f32(value) / 1_MB;
101 }
102
103 if (value <= 1_TB) {
104 suffix = "GB";
105 return f32(value) / 1_GB;
106 }
107
108 return f32(value);
109 }

111 void ImGuiDebugLayer::DrawStats() const {
112 ImGui::SetNextWindowPos({0, 0});
113 ImGui::Begin("Performance",
114 nullptr,
115 ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoDecoration | ImGuiWindowFlags_NoMove |
116 ImGuiWindowFlags_AlwaysAutoResize | ImGuiWindowFlags_NoBackground);
117
118 ImGui::Text("Frame Stats");
119 ImGui::Separator();
120 ImGui::TextColored(Colors::Green.To<ImVec4>(),
121 "Frame Rate %u FPS",
122 (u32)Math::Round(mFrameStats.frameRate));
123 ImGui::TextColored(Colors::Green.To<ImVec4>(), "Frame Time %.2f ms", mFrameStats.frameTime);
124 ImGui::TextColored(Colors::Magenta.To<ImVec4>(), "Main Thread %.2f ms", mFrameStats.mainThreadTime);
125 ImGui::TextColored(Colors::Magenta.To<ImVec4>(), "Render Thread %.2f ms", mFrameStats.renderThreadTime);
126 ImGui::TextColored(Colors::Cyan.To<ImVec4>(), "Draw Calls %u", mFrameStats.drawCalls);
127
128 ImGui::Dummy({0, 20.f});
129 ImGui::Text("Scene Stats");
130 ImGui::Separator();
131 ImGui::TextColored(Colors::Cyan.To<ImVec4>(), "Entities %u", mSceneStats.entities);
132
133 string allocatedSuffix, freeSuffix, usedSuffix;
134 const f32 allocatedOOM = CalcBytesOOM(mSceneStats.resourcePoolAllocatedBytes, allocatedSuffix);
135 const f32 freeOOM =
136 CalcBytesOOM(mSceneStats.resourcePoolAllocatedBytes - mSceneStats.resourcePoolUsedBytes, freeSuffix);
137 const f32 usedOOM = CalcBytesOOM(mSceneStats.resourcePoolUsedBytes, usedSuffix);
138
139 ImGui::TextColored(Colors::Yellow.To<ImVec4>(),
140 "Resource Pool (Allocated) %.1f %s",
141 allocatedOOM,
142 allocatedSuffix.c_str());
143 ImGui::TextColored(Colors::Yellow.To<ImVec4>(),
144 "Resource Pool (Free) %.1f %s",
145 freeOOM,
146 freeSuffix.c_str());
147 ImGui::TextColored(Colors::Yellow.To<ImVec4>(),
148 "Resource Pool (Used) %.1f %s",
149 usedOOM,
150 usedSuffix.c_str());
151
152 ImGui::End();
153 }
