32 FrameAllocator::FrameAllocator(size_t sizePerBuffer, u32 bufferCount)
33 : mBufferCount(bufferCount), mCurrentBuffer(0) {
34 mBuffers = new FrameBuffer[bufferCount];
35
36 for (u32 i = 0; i < bufferCount; ++i) {
37 mBuffers[i].memory = new u8[sizePerBuffer];
38 mBuffers[i].currentPos = mBuffers[i].memory;
39 mBuffers[i].size = sizePerBuffer;
40 mBuffers[i].peakUsage = 0;
41 }
42 }

44 FrameAllocator::~FrameAllocator() {
45 if (mBuffers) {
46 for (u32 i = 0; i < mBufferCount; ++i) {
47 delete[] mBuffers[i].memory;
48 }
49 delete[] mBuffers;
50 }
51 }

84 void* FrameAllocator::Allocate(size_t size, size_t alignment) {
85 FrameBuffer& buffer = mBuffers[mCurrentBuffer];
86
87 const auto currentAddr = RCAST<size_t>(buffer.currentPos);
88 const auto alignAddr = ASTERA_ALIGN_UP(currentAddr, alignment);
89 const auto adjustment = alignAddr - currentAddr;
90
91 // Check if we have enough space
92 if (buffer.currentPos + size + adjustment > buffer.memory + buffer.size) {
93 return nullptr; // Out of memory in the current frame buffer
94 }
95
96 // Update position
97 buffer.currentPos = RCAST<u8*>(alignAddr + size);
98
99 // Track peak usage
100 const size_t currentUsage = CAST<size_t>(buffer.currentPos - buffer.memory);
101 if (currentUsage > buffer.peakUsage) { buffer.peakUsage = currentUsage; }
102
103 return RCAST<void*>(alignAddr);
104 }

106 void FrameAllocator::NextFrame() {
107 // Move to the next buffer
108 mCurrentBuffer = (mCurrentBuffer + 1) % mBufferCount;
109
110 // Reset the new current buffer (it was used N frames ago)
111 FrameBuffer& buffer = mBuffers[mCurrentBuffer];
112 buffer.currentPos = buffer.memory;
113 buffer.peakUsage = 0;
114 }
