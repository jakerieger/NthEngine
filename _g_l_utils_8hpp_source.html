38 // Convert OpenGL error code to string
39 constexpr std::string_view GLErrorToString(GLenum error) {
40 switch (error) {
41 case GL_NO_ERROR:
42 return "GL_NO_ERROR";
43 case GL_INVALID_ENUM:
44 return "GL_INVALID_ENUM";
45 case GL_INVALID_VALUE:
46 return "GL_INVALID_VALUE";
47 case GL_INVALID_OPERATION:
48 return "GL_INVALID_OPERATION";
49 case GL_STACK_OVERFLOW:
50 return "GL_STACK_OVERFLOW";
51 case GL_STACK_UNDERFLOW:
52 return "GL_STACK_UNDERFLOW";
53 case GL_OUT_OF_MEMORY:
54 return "GL_OUT_OF_MEMORY";
55 case GL_INVALID_FRAMEBUFFER_OPERATION:
56 return "GL_INVALID_FRAMEBUFFER_OPERATION";
57 default:
58 return "UNKNOWN_ERROR";
59 }
60 }

62 // Check for OpenGL errors and report them
63 inline void GLCheckError(std::string_view function_name, const std::source_location& location) {
64 GLenum error;
65 bool hasError = false;
66
67 while ((error = glGetError()) != GL_NO_ERROR) {
68 hasError = true;
69 auto msg = fmt::format("[OpenGL Error] {}\n"
70 " Function: {}\n"
71 " File: {}:{}\n"
72 " In function: {}\n",
73 GLErrorToString(error),
74 function_name,
75 location.file_name(),
76 location.line(),
77 location.function_name());
78 fprintf(stderr, "%s\n", msg.c_str());
79 }
80
81 if (hasError) { throw OpenGLException(); }
82 }

84 // Helper to call and check - void version
85 template<typename Func, typename... Args>
86 requires std::is_void_v<std::invoke_result_t<Func, Args...>>
87 void GLCallImpl(const char* func_name, const std::source_location& loc, Func&& func, Args&&... args) {
88 while (glGetError() != GL_NO_ERROR)
89 ;
90 std::forward<Func>(func)(std::forward<Args>(args)...);
91 GLCheckError(func_name, loc);
92 }
93
94 // Helper to call and check - non-void version
95 template<typename Func, typename... Args>
96 requires(!std::is_void_v<std::invoke_result_t<Func, Args...>>)
97 auto GLCallImpl(const char* func_name, const std::source_location& loc, Func&& func, Args&&... args) {
98 while (glGetError() != GL_NO_ERROR)
99 ;
100 auto result = std::forward<Func>(func)(std::forward<Args>(args)...);
101 GLCheckError(func_name, loc);
102 return result;
103 }
104
105// Macro that works for both void and non-void
106#define GLCall(func, ...) GLCallImpl(#func, std::source_location::current(), func, ##__VA_ARGS__)
