32 Clock::Clock() {
33#ifdef ASTERA_PLATFORM_WINDOWS
34 QueryPerformanceFrequency(&mFrequency);
35 QueryPerformanceCounter(&mLastTime);
36#else
37 // Linux uses clock_gettime with CLOCK_MONOTONIC
38 mFrequency = 1000000000ULL; // nanoseconds per second
39
40 struct timespec ts {};
41 clock_gettime(CLOCK_MONOTONIC, &ts);
42 mLastTime = CAST<u64>(ts.tv_sec) * 1000000000ULL + CAST<u64>(ts.tv_nsec);
43#endif
44
45 mDeltaTime = 0.0;
46 mTotalTime = 0.0;
47 mFrameCount = 0;
48 mFramesPerSecond = 0.0;
49 mFpsUpdateInterval = 0.5; // In seconds
50 mTimeSinceLastFpsUpdate = 0.0;
51 }

53 void Clock::Tick() {
54#ifdef ASTERA_PLATFORM_WINDOWS
55 LARGE_INTEGER currentTime;
56 QueryPerformanceCounter(&currentTime);
57
58 mDeltaTime = CAST<f64>(currentTime.QuadPart - mLastTime.QuadPart) / CAST<f64>(mFrequency.QuadPart);
59 mLastTime = currentTime;
60#else
61 struct timespec ts {};
62 clock_gettime(CLOCK_MONOTONIC, &ts);
63 u64 currentTime = CAST<u64>(ts.tv_sec) * 1000000000ULL + CAST<u64>(ts.tv_nsec);
64
65 mDeltaTime = CAST<f64>(currentTime - mLastTime) / CAST<f64>(mFrequency);
66 mLastTime = currentTime;
67#endif
68
69 mTotalTime += mDeltaTime;
70 mFrameCount++;
71 mTimeSinceLastFpsUpdate += mDeltaTime;
72
73 if (mTimeSinceLastFpsUpdate >= mFpsUpdateInterval) {
74 mFramesPerSecond = CAST<f64>(mFrameCount) / mTimeSinceLastFpsUpdate;
75 mFrameCount = 0;
76 mTimeSinceLastFpsUpdate = 0.0;
77 }
78 }
