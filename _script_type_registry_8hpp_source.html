94 struct LuaTypeTraits<SceneState> {
95 static constexpr std::string_view typeName = "SceneState";
96
97 static void RegisterMembers(sol::usertype<SceneState>& usertype) {
98 usertype["FindEntityByName"] = [](SceneState& scene, const string& name) -> sol::optional<Entity> {
99 const auto view = scene.View<Transform>();
100 for (auto entity : view) {
101 if (scene.GetEntityName(entity) == name) { return entity; }
102 }
103 return sol::nullopt;
104 };
105
106 usertype["GetEntityTransform"] = [](SceneState& scene, Entity entity) -> Transform* {
107 auto& transform = scene.GetTransform(entity);
108 return &transform;
109 };
110 }
111 };
