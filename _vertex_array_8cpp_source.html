56 void VertexArray::AddVertexBuffer(const shared_ptr<VertexBuffer>& buffer, const VertexLayout& layout) {
57 ASTERA_ASSERT(mArrayID != 0);
58 ASTERA_ASSERT(buffer != nullptr);
59 ASTERA_ASSERT(!layout.GetAttributes().empty());
60
61 Bind();
62 buffer->Bind();
63
64 const auto& attributes = layout.GetAttributes();
65 u32 offset = 0;
66
67 for (const auto& attribute : attributes) {
68 GLCall(glEnableVertexAttribArray, mVertexBufferIndex);
69 GLCall(glVertexAttribPointer,
70 mVertexBufferIndex,
71 CAST<GLint>(attribute.GetComponentCount()),
72 attribute.GetGLType(),
73 attribute.normalized ? GL_TRUE : GL_FALSE,
74 CAST<GLsizei>(layout.GetStride()),
75 RCAST<void*>(CAST<uintptr_t>(offset)));
76
77 offset += attribute.GetSize();
78 mVertexBufferIndex++;
79 }
80
81 mVertexBuffers.push_back(buffer);
82 }

101 void VertexArray::Destroy() {
