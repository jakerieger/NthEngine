32 void Window::Run() {
33 mRunning = Initialize();
34 if (!mRunning)
35 return;
36
37 OnAwake();
38
39 while (mRunning && !glfwWindowShouldClose(mWindow)) {
40 mClock.Tick();
41 OnUpdate(mClock);
42 OnLateUpdate();
43 glfwPollEvents();
44 }
45
46 OnDestroyed();
47 Shutdown();
48 }

63 void Window::SetIcon(const Path& filename) {
64 i32 width, height, channels;
65 u8* pixels = stbi_load(filename.string().c_str(), &width, &height, &channels, 4);
66
67 if (pixels) {
68 GLFWimage icon;
69 icon.width = width;
70 icon.height = height;

74 } else {
75 Log::Error("Window", "Failed to load window icon: {}", filename.string());

79 bool Window::Initialize() {
80 if (!glfwInit()) {
81 Log::Critical("Window", "Failed to initialize GLFW");
82 return false;
83 }
84
85 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
86 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
87 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
88#ifdef __APPLE__
89 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
90#endif
91
92 mWindow = glfwCreateWindow((i32)mWidth, (i32)mHeight, mTitle.c_str(), nullptr, nullptr);
93 if (!mWindow) {
94 glfwTerminate();
95 Log::Critical("Window", "Failed to create GLFW window");
96 return false;
97 }
98
99 glfwMakeContextCurrent(mWindow);
100 glfwSetWindowUserPointer(mWindow, this);
101
102 // Set up callbacks
103 glfwSetFramebufferSizeCallback(mWindow, GLFWResizeCallback);
104 glfwSetKeyCallback(mWindow, GLFWKeyCallback);
105 glfwSetMouseButtonCallback(mWindow, GLFWMouseButtonCallback);
106 glfwSetCursorPosCallback(mWindow, GLFWMouseCursorCallback);
107 glfwSetScrollCallback(mWindow, GLFWMouseScrollCallback);
108
109 if (mVsync)
110 glfwSwapInterval(1);
111 else
112 glfwSwapInterval(0);
113
114 // Initialize OpenGL
115 if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress)) {
116 Log::Critical("Window", "Failed to load GLAD");
117 glfwDestroyWindow(mWindow);
118 glfwTerminate();
119 return false;
120 }
121
122 Log::Debug("Window", "Window initialized: {}x{}", mWidth, mHeight);
123 return true;
124 }

145 void Window::GLFWKeyCallback(GLFWwindow* window, i32 key, i32 scancode, i32 action, i32 mods) {
146 ASTERA_UNUSED(scancode);
147 ASTERA_UNUSED(mods);
148
149 auto* self = CAST<Window*>(glfwGetWindowUserPointer(window));
150 if (self) {
