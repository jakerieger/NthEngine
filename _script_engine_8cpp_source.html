49 void ScriptEngine::LoadScript(const string& source, ScriptID scriptId, ScriptType type) {
50 if (!mInitialized) {
51 PrintUninitializedError();
52 return;
53 }
54
55 try {
56 auto env = sol::environment(mLua, sol::create, mLua.globals());
57 mLua.script(source, env);
58
59 if (type == ScriptType::Behavior) {
60 sol::protected_function awakeFunc = env["OnAwake"];
61 sol::protected_function updateFunc = env["OnUpdate"];
62 sol::protected_function lateUpdateFunc = env["OnLateUpdate"];
63 sol::protected_function destroyedFunc = env["OnDestroyed"];
64
65 mBehaviorScriptContexts[scriptId] = {std::move(env),
66 std::move(awakeFunc),
67 std::move(updateFunc),
68 std::move(lateUpdateFunc),
69 std::move(destroyedFunc)};
70
71 Log::Debug("ScriptEngine", "Loaded script with id `{}`", scriptId);
72 }
73 } catch (const sol::error& e) { Log::Error("ScriptEngine", "Error loading script: {}", e.what()); }
74 }

76 void ScriptEngine::LoadScript(const vector<u8>& bytecode, ScriptID scriptId, const ScriptType type) {
77 if (!mInitialized) {
78 PrintUninitializedError();
79 return;
80 }
81
82 try {
83 auto env = sol::environment(mLua, sol::create, mLua.globals());
84
85 struct ReaderState {
86 const vector<u8>* bytecode {nullptr};
87 size_t position {0};
88 };
89
90 ReaderState state;
91 state.bytecode = &bytecode;
92
93 auto reader = [](lua_State* L, void* data, size_t* size) -> const char* {
94 const auto readerState = CAST<ReaderState*>(data);
95 if (readerState->position >= readerState->bytecode->size()) {
96 *size = 0;
97 return nullptr;
98 }
99
100 *size = readerState->bytecode->size() - readerState->position;
101 const auto result = RCAST<const char*>(readerState->bytecode->data() + readerState->position);
102 readerState->position = readerState->bytecode->size();
103 return result;
104 };
105
106 const auto chunkName = fmt::format("chunk_{}", scriptId);
107 sol::load_result loadedChunk = mLua.load(reader, &state, chunkName.c_str(), sol::load_mode::binary);
108 if (!loadedChunk.valid()) {
109 sol::error err = loadedChunk;
110 return;
111 }
112
113 if (const auto result = loadedChunk(env); !result.valid()) {
114 sol::error err = result;
115 return;
116 }
117
118 if (type == ScriptType::Behavior) {
119 sol::protected_function awakeFunc = env["OnAwake"];
120 sol::protected_function updateFunc = env["OnUpdate"];
121 sol::protected_function lateUpdateFunc = env["OnLateUpdate"];
122 sol::protected_function destroyedFunc = env["OnDestroyed"];
123
124 mBehaviorScriptContexts[scriptId] = {std::move(env),
125 std::move(awakeFunc),
126 std::move(updateFunc),
127 std::move(lateUpdateFunc),
128 std::move(destroyedFunc)};
129 }
130 } catch (const sol::error& e) { Log::Error("ScriptEngine", "Error loading script: {}", e.what()); }
131 }

133 void ScriptEngine::CallAwakeBehavior(const ScriptID id, const BehaviorEntity& entity) {
134 if (!mInitialized) {
135 PrintUninitializedError();
136 return;
137 }
138
139 if (!mBehaviorScriptContexts.contains(id)) {
140 PrintScriptNotFoundError(id);
141 return;
142 }
143
144 const auto& ctx = mBehaviorScriptContexts[id];
145 if (ctx.OnAwake.valid()) {
146 try {
147 std::ignore = ctx.OnAwake(entity);
148 } catch (const sol::error& e) { Log::Error("ScriptEngine", "{}", e.what()); }
149 }
