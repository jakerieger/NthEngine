34 GeometryHandle Geometry::CreateQuad(f32 width, f32 height) {
35 auto geometry = make_shared<Geometry>();
36
37 // Calculate half-extents for centered quad
38 const f32 halfWidth = width * 0.5f;
39 const f32 halfHeight = height * 0.5f;
40
41 // Define quad vertices (position + texcoord packed as vec4)
42 // Triangle strip ordering: bottom-left, bottom-right, top-left, top-right
43 const SpriteVertex vertices[] = {
44 {-halfWidth, -halfHeight, 0.0f, 0.0f}, // Bottom-left
45 {halfWidth, -halfHeight, 1.0f, 0.0f}, // Bottom-right
46 {-halfWidth, halfHeight, 0.0f, 1.0f}, // Top-left
47 {halfWidth, halfHeight, 1.0f, 1.0f}, // Top-right
48 };
49
50 // Define indices for two triangles
51 const u32 indices[] = {
52 0,
53 1,
54 2, // First triangle
55 2,
56 1,
57 3 // Second triangle
58 };
59
60 // Create vertex buffer and upload data
61 auto vertexBuffer = make_shared<VertexBuffer>();
62 vertexBuffer->SetData(vertices, sizeof(vertices), BufferUsage::Static);
63
64 // Create index buffer and upload data
65 auto indexBuffer = make_shared<IndexBuffer>();
66 indexBuffer->SetIndices(indices, 6, BufferUsage::Static);
67
68 // Define vertex layout matching the shader's input
69 // layout (location = 0) in vec4 aVertex;
70 VertexLayout layout;
71 layout.AddAttribute(VertexAttribute("aVertex", AttributeType::Float4));
72
73 // Create and configure VAO
74 geometry->mVertexArray = make_shared<VertexArray>();
75 geometry->mVertexArray->AddVertexBuffer(vertexBuffer, layout);
76 geometry->mVertexArray->SetIndexBuffer(indexBuffer);
77
78 Log::Debug("Geometry", "Created quad geometry ({}x{})", width, height);
79
80 return geometry;
81 }
