46 class TextureSprite {

49 public:
50 void Bind(u32 slot = 0) const {
51 ASTERA_ASSERT_MSG(slot <= 32, "Slot number too large. Slot must be <= 32");
52
53 if (mId != kInvalidTextureID) {
54 GLCall(glActiveTexture, GL_TEXTURE0 + slot);
55 GLCall(glBindTexture, GL_TEXTURE_2D, mId);
56 }
57 }
58
59 void Unbind() const {
60 if (mId != kInvalidTextureID) {
61 GLCall(glBindTexture, GL_TEXTURE_2D, 0);
62 }
63 }
