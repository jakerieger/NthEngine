28 ASTERA_KEEP constexpr std::string_view TrimLeading(std::string_view sv) noexcept {
29 while (!sv.empty() &&
30 (sv.front() == ' ' || sv.front() == '\t' || sv.front() == '\n' || sv.front() == '\r')) {
31 sv.remove_prefix(1);
32 }
33 return sv;
34 }

37 template<typename T>
38 requires std::is_integral_v<T>
39 ASTERA_KEEP Result<T> ParseInteger(std::string_view str, int base = 10) noexcept {
40 str = TrimLeading(str);
41
42 if (str.empty()) {
43 return unexpected("Empty input string");
44 }
45
46 T result {};
47 const char* first = str.data();
48 const char* last = first + str.size();
49
50 auto [ptr, ec] = std::from_chars(first, last, result, base);
51
52 if (ec == std::errc::invalid_argument) {
53 return unexpected("Invalid integer format");
54 }
55 if (ec == std::errc::result_out_of_range) {
56 return unexpected("Integer value out of range");
57 }
58 if (ptr != last) {
59 std::string_view remaining(ptr, CAST<std::size_t>(last - ptr));
60 remaining = TrimLeading(remaining);
61 if (!remaining.empty()) {
62 return unexpected("Trailing characters in string");
63 }
64 }
65
66 return result;
67 }

104 template<typename T>
105 constexpr std::size_t MaxCharsSize() noexcept {
106 if constexpr (std::is_same_v<T, i32>) {
107 return 12; // -2147483648
108 } else if constexpr (std::is_same_v<T, u32>) {
109 return 11; // 4294967295
110 } else if constexpr (std::is_same_v<T, i64>) {
111 return 21; // -9223372036854775808
112 } else if constexpr (std::is_same_v<T, u64>) {
113 return 21; // 18446744073709551615
114 } else if constexpr (std::is_same_v<T, f32>) {
115 return 48;
116 } else if constexpr (std::is_same_v<T, f64>) {
117 return 328;
118 } else {
119 return 64;
120 }
121 }
