51 inline static Vec2 ScreenToNDC(const Vec2& screenPos, f32 screenWidth, f32 screenHeight) {
52 const f32 ndcX = (screenPos.x / screenWidth) * 2.0f - 1.0f;
53 const f32 ndcY = (screenPos.y / screenHeight) * 2.0f - 1.0f;
54 return {ndcX, ndcY};
55 }

62 inline static Vec2 NDCToScreen(const Vec2& ndcPos, f32 screenWidth, f32 screenHeight) {
63 const f32 screenX = (ndcPos.x + 1.0f) * 0.5f * screenWidth;
64 const f32 screenY = (ndcPos.y + 1.0f) * 0.5f * screenHeight;
65 return {screenX, screenY};
66 }

77 inline static Vec2 ScreenToWorld(const Vec2& screenPos,
78 f32 screenWidth,
79 f32 screenHeight,
80 f32 worldLeft = 0.0f,
81 f32 worldRight = 0.0f,
82 f32 worldBottom = 0.0f,
83 f32 worldTop = 0.0f) {
84 // Default to 1:1 mapping if world bounds not specified
85 if (worldRight == 0.0f) worldRight = screenWidth;
86 if (worldTop == 0.0f) worldTop = screenHeight;
87
88 const f32 worldX = worldLeft + (screenPos.x / screenWidth) * (worldRight - worldLeft);
89 const f32 worldY = worldBottom + (screenPos.y / screenHeight) * (worldTop - worldBottom);
90 return {worldX, worldY};
91 }

102 inline static Vec2 WorldToScreen(const Vec2& worldPos,
103 f32 screenWidth,
104 f32 screenHeight,
105 f32 worldLeft = 0.0f,
106 f32 worldRight = 0.0f,
107 f32 worldBottom = 0.0f,
108 f32 worldTop = 0.0f) {
109 // Default to 1:1 mapping if world bounds not specified
110 if (worldRight == 0.0f) worldRight = screenWidth;
111 if (worldTop == 0.0f) worldTop = screenHeight;
112
113 const f32 screenX = ((worldPos.x - worldLeft) / (worldRight - worldLeft)) * screenWidth;
114 const f32 screenY = ((worldPos.y - worldBottom) / (worldTop - worldBottom)) * screenHeight;
115 return {screenX, screenY};
116 }

123 inline static Vec2 NormalizedToScreen(const Vec2& normalizedPos, f32 screenWidth, f32 screenHeight) {

154 inline static Mat4 CreateScreenProjection(f32 screenWidth, f32 screenHeight, bool originTopLeft = false) {
155 if (originTopLeft) {
156 // Top-left origin (common for UI and 2D games)
157 return glm::ortho(0.0f, screenWidth, screenHeight, 0.0f, -1.0f, 1.0f);
158 } else {
159 // Bottom-left origin (OpenGL default)
160 return glm::ortho(0.0f, screenWidth, 0.0f, screenHeight, -1.0f, 1.0f);
161 }
162 }

168 inline static f32 GetAspectRatio(f32 screenWidth, f32 screenHeight) {

177 inline static Vec2 ClampToScreen(const Vec2& pos, f32 screenWidth, f32 screenHeight) {
178 return {glm::clamp(pos.x, 0.0f, screenWidth), glm::clamp(pos.y, 0.0f, screenHeight)};
179 }

186 inline static bool IsOnScreen(const Vec2& pos, f32 screenWidth, f32 screenHeight) {
187 return pos.x >= 0.0f && pos.x <= screenWidth && pos.y >= 0.0f && pos.y <= screenHeight;
188 }

196 inline static Vec2
197 MouseToWorld(const Vec2& mousePos, f32 screenWidth, f32 screenHeight, bool originTopLeft = false) {
198 if (originTopLeft) {
199 // Mouse coords typically have Y down, and our world does too
200 return mousePos;
201 } else {
202 // Mouse Y is down, but our world Y is up, so flip it
203 return {mousePos.x, screenHeight - mousePos.y};
204 }
205 }
