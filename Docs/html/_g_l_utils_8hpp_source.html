38 // Check for OpenGL errors and report them
39 inline void GLCheckError(std::string_view function_name, const std::source_location& location) {
40 GLenum error;
41 bool hasError = false;
42
43 while ((error = glGetError()) != GL_NO_ERROR) {
44 hasError = true;
45 auto msg = fmt::format("[OpenGL Error] {}\n"
46 " Function: {}\n"
47 " File: {}:{}\n"
48 " In function: {}\n",
49 GLErrorToString(error),
50 function_name,
51 location.file_name(),
52 location.line(),
53 location.function_name());
54 fprintf(stderr, "%s\n", msg.c_str());
55 }
56
57 if (hasError) { throw OpenGLException(); }
58 }
