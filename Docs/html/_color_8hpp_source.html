92 template<>
93 N_ND inline u32 Color::To() const {
94 const auto redByte = CAST<u8>(FloatToU32(mRed));
95 const auto greenByte = CAST<u8>(FloatToU32(mGreen));
96 const auto blueByte = CAST<u8>(FloatToU32(mBlue));
97 const auto alphaByte = CAST<u8>(FloatToU32(mAlpha));
98 return (alphaByte << 24) | (redByte << 16) | (greenByte << 8) | blueByte;
99 }

101 template<>
102 N_ND inline string Color::To() const {
103 const u32 r = U32ToFloat(mRed + 0.5f);
104 const u32 g = U32ToFloat(mGreen + 0.5f);
105 const u32 b = U32ToFloat(mBlue + 0.5f);
106 std::stringstream ss;
107 ss << "#" << std::hex << std::setfill('0') << std::setw(2) << (r & 0xFF) << std::setw(2) << (g & 0xFF)
108 << std::setw(2) << (b & 0xFF);
109 return ss.str();
110 }
