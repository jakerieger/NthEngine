95 void Shader::CompileShaders(const char* vertexSource, const char* fragSource) {
96 const u32 vertexShader = GLCall(glCreateShader, GL_VERTEX_SHADER);
97 GLCall(glShaderSource, vertexShader, 1, &vertexSource, nullptr);
98 GLCall(glCompileShader, vertexShader);
99
100 // CHECK VERTEX SHADER COMPILATION
101 GLint success;
102 glGetShaderiv(vertexShader, GL_COMPILE_STATUS, &success);
103 if (!success) {
104 GLchar infoLog[512];
105 glGetShaderInfoLog(vertexShader, 512, nullptr, infoLog);
106 Log::Error("Shader", "Vertex shader compilation failed: {}", infoLog);
107 }
108
109 const u32 fragmentShader = GLCall(glCreateShader, GL_FRAGMENT_SHADER);
110 GLCall(glShaderSource, fragmentShader, 1, &fragSource, nullptr);
111 GLCall(glCompileShader, fragmentShader);
112
113 // CHECK FRAGMENT SHADER COMPILATION
114 glGetShaderiv(fragmentShader, GL_COMPILE_STATUS, &success);
115 if (!success) {
116 GLchar infoLog[512];
117 glGetShaderInfoLog(fragmentShader, 512, nullptr, infoLog);
118 Log::Error("Shader", "Fragment shader compilation failed: {}", infoLog);
119 }
120
121 mProgram = GLCall(glCreateProgram);
122 GLCall(glAttachShader, mProgram, vertexShader);
123 GLCall(glAttachShader, mProgram, fragmentShader);
124 GLCall(glLinkProgram, mProgram);
125
126 // CHECK PROGRAM LINKING
127 glGetProgramiv(mProgram, GL_LINK_STATUS, &success);
128 if (!success) {
129 GLchar infoLog[512];
130 glGetProgramInfoLog(mProgram, 512, nullptr, infoLog);
131 Log::Error("Shader", "Shader program linking failed: {}", infoLog);
132 }
133
134 GLCall(glValidateProgram, mProgram);
135
136 // CHECK PROGRAM VALIDATION
137 glGetProgramiv(mProgram, GL_VALIDATE_STATUS, &success);
138 if (!success) {
139 GLchar infoLog[512];
140 glGetProgramInfoLog(mProgram, 512, nullptr, infoLog);
141 Log::Error("Shader", "Shader program validation failed: {}", infoLog);
142 }
143
144 GLCall(glDeleteShader, vertexShader);
145 GLCall(glDeleteShader, fragmentShader);
146 }
