51 inline static Vec2 ScreenToWorld(const Vec2& screenPos,
52 f32 screenWidth,
53 f32 screenHeight,
54 f32 worldLeft = 0.0f,
55 f32 worldRight = 0.0f,
56 f32 worldBottom = 0.0f,
57 f32 worldTop = 0.0f) {
58 // Default to 1:1 mapping if world bounds not specified
59 if (worldRight == 0.0f) worldRight = screenWidth;
60 if (worldTop == 0.0f) worldTop = screenHeight;
61
62 const f32 worldX = worldLeft + (screenPos.x / screenWidth) * (worldRight - worldLeft);
63 const f32 worldY = worldBottom + (screenPos.y / screenHeight) * (worldTop - worldBottom);
64 return {worldX, worldY};
