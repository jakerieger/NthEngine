53 inline static Vec2 ScreenToWorld(const Vec2& screenPos,
54 f32 screenWidth,
55 f32 screenHeight,
56 f32 worldLeft = 0.0f,
57 f32 worldRight = 0.0f,
58 f32 worldBottom = 0.0f,
59 f32 worldTop = 0.0f) {
60 // Default to 1:1 mapping if world bounds not specified
61 if (worldRight == 0.0f) worldRight = screenWidth;
62 if (worldTop == 0.0f) worldTop = screenHeight;
63
64 const f32 worldX = worldLeft + (screenPos.x / screenWidth) * (worldRight - worldLeft);
65 const f32 worldY = worldBottom + (screenPos.y / screenHeight) * (worldTop - worldBottom);
66 return {worldX, worldY};
67 }

78 inline static Vec2 WorldToScreen(const Vec2& worldPos,
79 f32 screenWidth,
80 f32 screenHeight,
81 f32 worldLeft = 0.0f,
82 f32 worldRight = 0.0f,
83 f32 worldBottom = 0.0f,
84 f32 worldTop = 0.0f) {
85 // Default to 1:1 mapping if world bounds not specified
86 if (worldRight == 0.0f) worldRight = screenWidth;
87 if (worldTop == 0.0f) worldTop = screenHeight;
88
89 const f32 screenX = ((worldPos.x - worldLeft) / (worldRight - worldLeft)) * screenWidth;
90 const f32 screenY = ((worldPos.y - worldBottom) / (worldTop - worldBottom)) * screenHeight;

130 inline static Mat4 CreateScreenProjection(f32 screenWidth, f32 screenHeight, bool originTopLeft = false) {
131 if (originTopLeft) {
132 // Top-left origin (common for UI and 2D games)
133 return glm::ortho(0.0f, screenWidth, screenHeight, 0.0f, -1.0f, 1.0f);

172 inline static Vec2
173 MouseToWorld(const Vec2& mousePos, f32 screenWidth, f32 screenHeight, bool originTopLeft = false) {
174 if (originTopLeft) {
175 // Mouse coords typically have Y down, and our world does too
176 return mousePos;
177 } else {
178 // Mouse Y is down, but our world Y is up, so flip it
179 return {mousePos.x, screenHeight - mousePos.y};
180 }
