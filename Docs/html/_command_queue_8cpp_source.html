76 void CommandExecutor::operator()(const SetUniformCommand& cmd) const {

78
79 const GLint location = glGetUniformLocation(cmd.programId, cmd.name.c_str());
80 if (location == -1) {
81 Log::Warn("CommandQueue", "Uniform '{}' not found in shader program {}", cmd.name, cmd.programId);
82 return;
83 }

85 // Use std::visit to handle the different uniform value types
86 std::visit(
87 [location]<typename T0>(const T0& value) {
88 using T = std::decay_t<T0>;
89
90 if constexpr (std::is_same_v<T, i32>) {
91 GLCall(glUniform1i, location, value);
92 } else if constexpr (std::is_same_v<T, f32>) {
93 GLCall(glUniform1f, location, value);
94 } else if constexpr (std::is_same_v<T, glm::vec2>) {
95 GLCall(glUniform2f, location, value.x, value.y);
96 } else if constexpr (std::is_same_v<T, glm::vec3>) {
97 GLCall(glUniform3f, location, value.x, value.y, value.z);
98 } else if constexpr (std::is_same_v<T, glm::vec4>) {
99 GLCall(glUniform4f, location, value.x, value.y, value.z, value.w);
100 } else if constexpr (std::is_same_v<T, glm::mat4>) {
101 GLCall(glUniformMatrix4fv, location, 1, GL_FALSE, &value[0][0]);

103 },
