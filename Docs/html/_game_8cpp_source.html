18 void Game::Run() {

128 bool Game::Initialize() {
129 // Enforce X11 on Wayland systems for debugging support (RenderDoc doens't support Wayland)
130#if defined(N_ENGINE_PLATFORM_LINUX_WAYLAND) || defined(N_ENGINE_PLATFORM_LINUX_X11)
131 glfwInitHint(GLFW_PLATFORM, GLFW_PLATFORM_X11);
132#endif
133
134 if (!glfwInit()) {
135 Log::Critical("Game", "Failed to initialize GLFW");
136 return false;
137 }
138
139 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
140 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 6);
141 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
142#ifdef __APPLE__
143 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
144#endif
145
146 mWindow = glfwCreateWindow((i32)mWidth, (i32)mHeight, mTitle.c_str(), nullptr, nullptr);
147 if (!mWindow) {
148 glfwTerminate();
149 Log::Critical("Game", "Failed to create GLFW mWindow");
150 return false;
151 }
152
153 glfwMakeContextCurrent(mWindow);
154 glfwSetWindowUserPointer(mWindow, this);
155
156 glfwSetFramebufferSizeCallback(mWindow, GLFWResizeCallback);
157 glfwSetKeyCallback(mWindow, GLFWKeyCallback);
158 glfwSetMouseButtonCallback(mWindow, GLFWMouseButtonCallback);
159 glfwSetCursorPosCallback(mWindow, GLFWMouseCursorCallback);
160 glfwSetScrollCallback(mWindow, GLFWMouseScrollCallback);
161
162 if (mVsync) glfwSwapInterval(1);
163 else glfwSwapInterval(0);
164
165 if (!mRenderContext.Initialize(mWidth, mHeight)) {
166 glfwDestroyWindow(mWindow);
167 glfwTerminate();
168 Log::Critical("Game", "Failed to initialize render context");
169 return false;
170 }
171
172 TextureManager::Initialize();
173 ShaderManager::Initialize();
174 mAudioEngine.Initialize();
175 InitializeScriptEngine();
176
177 // Debug layers
178 mImGuiDebugLayer = make_unique<ImGuiDebugLayer>(GetWindowHandle());
179 mDebugManager.AttachOverlay(mImGuiDebugLayer.get());
180
181 mPhysicsDebugLayer = make_unique<PhysicsDebugLayer>(mWidth, mHeight);
182 mDebugManager.AttachOverlay(mPhysicsDebugLayer.get());
183
184 Log::Debug("Game",
185 "Successfully initialized game instance:\n-- Dimensions: {}x{}\n-- V-Sync: {}",
186 mWidth,
187 mHeight,
188 mVsync ? "On" : "Off");
189
190 return true;
191 }

209 mScriptEngine.Initialize();
210 if (!mScriptEngine.IsInitialized()) return false;
211
212 // Register globals
213 auto& lua = mScriptEngine.GetLuaState();
214 auto gameGlobal = lua.new_usertype<Game>("Game");
215 gameGlobal["Quit"] = [this] { Quit(); };
216 gameGlobal["GetScreenSize"] = [this]() -> Vec2 { return {mWidth, mHeight}; };
217
218 // Register globals
219 Log::RegisterLuaGlobals(lua);
220 Math::RegisterLuaGlobals(lua);

222 mInputManager.RegisterLuaGlobals(lua);
223 mAudioEngine.RegisterLuaGlobals(lua);
224
225 // Register types
226 mScriptEngine.RegisterTypes<BehaviorEntity, Vec2, Clock, Transform>();
227
