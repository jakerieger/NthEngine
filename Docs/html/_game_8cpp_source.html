72#if defined(N_ENGINE_PLATFORM_LINUX_WAYLAND) || defined(N_ENGINE_PLATFORM_LINUX_X11)

98 glfwSetFramebufferSizeCallback(mWindow, GLFWResizeCallback);
99 glfwSetKeyCallback(mWindow, GLFWKeyCallback);
100 glfwSetMouseButtonCallback(mWindow, GLFWMouseButtonCallback);
101 glfwSetCursorPosCallback(mWindow, GLFWMouseCursorCallback);
102 glfwSetScrollCallback(mWindow, GLFWMouseScrollCallback);
103
104 if (mVsync) glfwSwapInterval(1);
105 else glfwSwapInterval(0);
106
107 if (!mRenderContext.Initialize(mWidth, mHeight)) {

115 ShaderManager::Initialize();
116
117 Log::Debug("Game",
118 "Successfully initialized game instance:\n-- Dimensions: {}x{}\n-- V-Sync: {}",
119 mWidth,
120 mHeight,
121 mVsync ? "On" : "Off");
122
123 return true;
