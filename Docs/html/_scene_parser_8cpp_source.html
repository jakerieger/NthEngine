18 const pugi::xml_node posNode = transformNode.child("Position");

46 static void ParseEntity(const pugi::xml_node& entityNode, EntityDescriptor& entity) {
47 entity.id = entityNode.attribute("id").as_int();
48 entity.name = entityNode.attribute("name").as_string();
49
50 const pugi::xml_node componentsNode = entityNode.child("Components");
51 if (!componentsNode) { throw std::runtime_error("Entity is missing Transform component."); }
52
53 // Parse Transform component
54 const pugi::xml_node transformNode = componentsNode.child("Transform");
55 if (transformNode) { entity.transform = ParseTransformComponent(transformNode); }
56
57 // Parse SpriteRenderer component
58 const pugi::xml_node spriteNode = componentsNode.child("SpriteRenderer");
59 if (spriteNode) { entity.spriteRenderer = ParseSpriteRendererComponent(spriteNode); }
60 }

66 void SceneParser::DescriptorToState(const SceneDescriptor& descriptor, SceneState& outState) {
67 auto entities = descriptor.entities;
68 std::ranges::sort(entities, {}, &EntityDescriptor::id); // Sort by id to ensure correct insertion order
69 for (const auto& entity : entities) {
70 const auto newEntity = outState.CreateEntity();
71 auto& transform = outState.GetTransform(newEntity);
72 transform.position = entity.transform.position;
73 transform.rotation = entity.transform.rotation;
74 transform.scale = entity.transform.scale;
75
76 if (entity.spriteRenderer.has_value()) {
77 auto& spriteRenderer = outState.AddComponent<SpriteRenderer>(newEntity);
78 spriteRenderer.geometry = Geometry::CreateQuad();
79 spriteRenderer.textureId =
80 TextureManager::Load(Content::GetContentPath(entity.spriteRenderer->texture));
81 }
82 }

89 void SceneParser::DeserializeDescriptor(const fs::path& filename, SceneDescriptor& outDescriptor) {
90 pugi::xml_document doc;
91 const pugi::xml_parse_result result = doc.load_file(filename.c_str());
92
93 if (!result) { throw std::runtime_error(fmt::format("XML parsing error: {}", result.description())); }
94
95 const pugi::xml_node sceneNode = doc.child("Scene");
96 if (!sceneNode) { throw std::runtime_error("No scene node found"); }
97
98 // Parse scene name
99 outDescriptor.name = sceneNode.attribute("name").as_string();
100
101 // Parse entities
102 const pugi::xml_node entitiesNode = sceneNode.child("Entities");
103 if (entitiesNode) {
104 for (pugi::xml_node entityNode : entitiesNode.children("Entity")) {
105 EntityDescriptor entity {};
106 ParseEntity(entityNode, entity);
107 outDescriptor.entities.push_back(entity);
108 }
109 }
