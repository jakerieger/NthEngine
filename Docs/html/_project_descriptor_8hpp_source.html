19 inline string ExpandMacros(const string& input) {
20 std::string result = input;
21 const std::regex macro_pattern(R"(\$\{([^}]+)\})");
22 std::smatch match;
23
24 while (std::regex_search(result, match, macro_pattern)) {
25 std::string macro_name = match[1].str();
26
27 auto it = mMacros.find(macro_name);
28 if (it != mMacros.end()) {
29 result.replace(match.position(), match.length(), it->second);
30 } else {
31 throw std::runtime_error("Macro not found");
32 }
33 }
34
35 return result;
36 }
37
38 inline void ExpandNode(pugi::xml_node node) {
39 if (node.type() == pugi::node_pcdata || node.type() == pugi::node_cdata) {
40 const string expanded = ExpandMacros(node.value());
41 node.set_value(expanded.c_str());
42 }
43
44 for (pugi::xml_attribute attr : node.attributes()) {
45 string expanded = ExpandMacros(attr.value());
46 attr.set_value(expanded.c_str());
47 }
48
49 for (pugi::xml_node child : node.children()) {
50 ExpandNode(child);
51 }
52 }
53 };
