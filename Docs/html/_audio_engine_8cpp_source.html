34 void AudioEngine::Shutdown() {

36
37 for (const auto& pair : mSounds) {

48 SoundID AudioEngine::LoadSound(const fs::path& filename) {
49 if (!mInitialized) { Initialize(); }
50
51 if (!exists(filename)) {
52 Log::Error("AudioEngine", "Attempted to load missing file: {}", filename.string());
53 return kInvalidSoundID;
54 }
55
56 auto sound = std::make_unique<Sound>();
57 sound->filepath = filename.string();
58
59 const ma_result result =
60 ma_sound_init_from_file(&mEngine, filename.string().c_str(), 0, nullptr, nullptr, &sound->sound);
61
62 if (result != MA_SUCCESS) {
63 Log::Error("AudioEngine", "Failed to load sound: {} (error: {})", filename.string(), (i32)result);
64 return kInvalidSoundID;
65 }
66
67 sound->loaded = true;
68 i32 soundId = mNextSoundId++;
69 mSounds[soundId] = std::move(sound);
70
71 Log::Info("AudioEngine", "Loaded sound: {} (ID: {})", filename.string(), soundId);
72 return soundId;
73 }

75 bool AudioEngine::PlaySound(SoundID id, bool loop) {
76 auto* sound = GetSound(id);
77 if (!sound) {
78 Log::Error("AudioEngine", "Sound ID {} not found", id);
79 return false;
80 }
81
82 ma_sound_set_looping(&sound->sound, loop ? MA_TRUE : MA_FALSE);
83
84 ma_sound_seek_to_pcm_frame(&sound->sound, 0);
85
86 const ma_result result = ma_sound_start(&sound->sound);
87 if (result != MA_SUCCESS) {
88 Log::Error("AudioEngine", "Failed to play sound ID {} (error: {})", id, (i32)result);
89 return false;
90 }
91
92 return true;
93 }
