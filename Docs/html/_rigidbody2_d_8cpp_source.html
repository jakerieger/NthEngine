12 void Rigidbody2D::UpdateMass(f32 newMass) {
13 mass = newMass;
14 if (type == BodyType::Static) {
15 inverseMass = 0.0f;
16 inverseInertia = 0.0f;
17 } else {
18 inverseMass = (mass > 0.0f) ? 1.0f / mass : 0.0f;
19 // Simple inertia approximation for a point mass
20 // This should be recalculated based on collider shape
21 inertia = mass;
22 inverseInertia = (inertia > 0.0f) ? 1.0f / inertia : 0.0f;
23 }
24 }

36 void Rigidbody2D::ApplyImpulseAtPoint(const Vec2& impulse, const Vec2& contactPoint, const Vec2& centerOfMass) {
37 if (type != BodyType::Dynamic) return;
38
39 // Linear impulse
40 velocity += impulse * inverseMass;
41
42 // Angular impulse
43 if (!lockRotation) {
44 const Vec2 r = contactPoint - centerOfMass;
45 const auto cross = r.x * impulse.y - r.y * impulse.x;
46 angularVelocity += cross * inverseInertia;
47 }
