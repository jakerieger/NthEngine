35 class SoundLoader final : public ResourceLoader<Sound> {
36 Sound LoadImpl(RenderContext& context, ArenaAllocator& allocator, const u64 id) override {
37 const auto bytes = AssetManager::GetAssetData(id);
38 if (!bytes.has_value()) {
39 throw std::runtime_error(fmt::format("Failed to get sound asset: {}", bytes.error()));
40 }
41
42 auto* arenaData = (u8*)allocator.Allocate(bytes->size(), alignof(u8));
43 std::memcpy(arenaData, bytes->data(), bytes->size());
44
45 return Sound {arenaData, bytes->size()};
46 }
47 };
