210 template<typename Func>
211 void ParallelFor(size_t start, size_t end, Func func, size_t chunkSize = 0) {
212 if (!gJobSystem || !gJobSystem->IsInitialized()) {
213 // Fallback to serial execution
214 for (size_t i = start; i < end; ++i) {
215 func(i);
216 }
217 return;
218 }
219
220 const size_t count = end - start;
221 if (count == 0)
222 return;
223
224 // Auto-determine chunk size
225 if (chunkSize == 0) {
226 const size_t numWorkers = gJobSystem->GetWorkerCount();
227 chunkSize = Math::Max(size_t(1), count / (numWorkers * 4));
228 }
229
230 vector<JobSystem::Job> jobs;
231 for (size_t i = start; i < end; i += chunkSize) {
232 size_t chunkEnd = Math::Min(i + chunkSize, end);
233 jobs.push_back([i, chunkEnd, &func]() {
234 for (size_t j = i; j < chunkEnd; ++j) {
235 func(j);
236 }
237 });
238 }
239
240 auto counter = gJobSystem->SubmitBatch(jobs);
241 gJobSystem->WaitForCounter(counter);
242 }

246 template<typename Func>
247 void ParallelForIndexed(size_t start, size_t end, Func func, size_t chunkSize = 0) {
248 if (!gJobSystem || !gJobSystem->IsInitialized()) {
249 for (size_t i = start; i < end; ++i) {
250 func(i, 0);
251 }
252 return;
253 }
254
255 const size_t count = end - start;
256 if (count == 0)
257 return;
258
259 if (chunkSize == 0) {
260 const size_t numWorkers = gJobSystem->GetWorkerCount();
261 chunkSize = Math::Max(size_t(1), count / (numWorkers * 4));
262 }
263
264 vector<JobSystem::Job> jobs;
265 for (size_t i = start; i < end; i += chunkSize) {
266 size_t chunkEnd = Math::Min(i + chunkSize, end);
267 jobs.push_back([i, chunkEnd, &func]() {
268 i32 workerID = gJobSystem->GetCurrentWorkerID();
269 size_t workerIndex = workerID >= 0 ? static_cast<size_t>(workerID) : 0;
270 for (size_t j = i; j < chunkEnd; ++j) {
271 func(j, workerIndex);
272 }
273 });
274 }
275
276 auto counter = gJobSystem->SubmitBatch(jobs);
277 gJobSystem->WaitForCounter(counter);
278 }
