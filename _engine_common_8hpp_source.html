150 inline f32 StringToF32(const string& str) {
151 f32 result;
152 auto [ptr, ec] = std::from_chars(str.data(), str.data() + str.size(), result);
153
154 if (ec == std::errc()) {
155 // Success - also check if entire string was consumed
156 if (ptr == str.data() + str.size()) {
157 return result;
158 } else {
159 // Partial parse - trailing characters exist
160 throw std::invalid_argument("Trailing characters in string");
161 }
162 } else if (ec == std::errc::invalid_argument) {
163 throw std::invalid_argument("Invalid float format");
164 } else if (ec == std::errc::result_out_of_range) {
165 throw std::range_error("Float value out of range");
166 }
167
168 throw std::runtime_error("Unknown conversion error");
169 }
