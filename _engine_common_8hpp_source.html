157 inline f32 StringToF32(const string& str) {
158 f32 result;
159 auto [ptr, ec] = std::from_chars(str.data(), str.data() + str.size(), result);
160
161 if (ec == std::errc()) {
162 // Success - also check if the entire string was consumed
163 if (ptr == str.data() + str.size()) {
164 return result;
165 } else {
166 // Partial parse: trailing characters exist
167 throw std::invalid_argument("Trailing characters in string");
168 }
169 } else if (ec == std::errc::invalid_argument) {
170 throw std::invalid_argument("Invalid float format");
171 } else if (ec == std::errc::result_out_of_range) {
172 throw std::range_error("Float value out of range");
173 }
174
175 throw std::runtime_error("Unknown conversion error");
176 }
