95 template<typename... Loaders>
96 void RegisterLoaders() {
97 (
98 [this]() {
99 auto loader = make_unique<Loaders>();
100 mLoaders[std::type_index(typeid(typename Loaders::ResourceType))] = std::move(loader);
101 }(),
102 ...);
103 }
104
105 template<typename T>
106 bool LoadResource(const u64 id) {
107 if (mResources.contains(id)) {
108 return true; // asset already exists
109 }
110
111 const auto loaderIt = mLoaders.find(std::type_index(typeid(T)));
112 if (loaderIt == mLoaders.end()) {
113 return false; // no registered loader for type
114 }
115
116 ResourceBase* resource = loaderIt->second->Load(mRenderContext, mAllocator, id);
117 if (!resource) {
118 return false; // loading failed
119 }
120
121 mResources[id] = resource;
122 return true;
123 }
124
125 template<typename T>
126 ResourceHandle<T> FetchResource(const u64 id) {
127 const auto it = mResources.find(id);
128 if (it == mResources.end()) {
129 return ResourceHandle<T> {};
130 }
131
132 Resource<T>* typedResource = DCAST<Resource<T>*>(it->second);
133 if (!typedResource) {
134 return {};
135 }
136
137 return ResourceHandle<T>(this, id, &typedResource->data);
138 }
139
140 void Clear() {
141 mResources.clear();
142 mAllocator.Reset();
143 }
144
145 const ArenaAllocator& GetAllocator() {
146 return mAllocator;
147 }
148 };
