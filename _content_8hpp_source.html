48 class Content {
50 static constexpr std::string_view kSceneRoot = "Scenes";
52 static constexpr std::string_view kScriptRoot = "Scripts";
54 static constexpr std::string_view kShaderRoot = "Shaders";
56 static constexpr std::string_view kSpriteRoot = "Sprites";

61 inline static Path sContentRoot;
62
67 inline static Path sEngineContentRoot {ASTERA_ENGINE_CONTENT_DIR};
68
69 public:
76 template<ContentType type, bool engine = false>
77 inline static Path Get(const string& filename) {
78 Path contentPath = sContentRoot;
79 if constexpr (engine) { contentPath = sEngineContentRoot; }
80
81 if constexpr (type == ContentType::Audio) {
82 return contentPath / kAudioRoot / filename;
83 } else if constexpr (type == ContentType::Scene) {
84 return contentPath / kSceneRoot / filename;
85 } else if constexpr (type == ContentType::Script) {
86 return contentPath / kScriptRoot / filename;
87 } else if constexpr (type == ContentType::Shader) {
88 return contentPath / kShaderRoot / filename;
89 } else if constexpr (type == ContentType::Sprite) {
90 return contentPath / kSpriteRoot / filename;
91 } else {
92 throw std::invalid_argument("Invalid content type");
93 }
94 }
95

99 sContentRoot = path;
100 }
101
104 inline static void SetEngineContentPath(const Path& path) {
105 sEngineContentRoot = path;
106 }
107
109 inline static void SetRuntimeDefaults() {
110 sContentRoot = fs::current_path() / "Content";
111 sEngineContentRoot = sContentRoot / "EngineContent";
112 }
113 };
