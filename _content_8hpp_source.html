44 class Content {
46 static constexpr std::string_view kSceneRoot = "Scenes";
48 static constexpr std::string_view kScriptRoot = "Scripts";
50 static constexpr std::string_view kShaderRoot = "Shaders";
52 static constexpr std::string_view kSpriteRoot = "Sprites";
54 static constexpr std::string_view kAudioRoot = "Audio";
55
57 inline static fs::path sContentRoot;
59 inline static fs::path sEngineContentRoot;
60
61 public:
68 template<ContentType type, bool engine = false>
69 inline static fs::path Get(const string& filename) {
70 fs::path contentPath = sContentRoot;
71 if constexpr (engine) { contentPath = sEngineContentRoot; }
72
73 if constexpr (type == ContentType::Audio) {
74 return contentPath / kAudioRoot / filename;
75 } else if constexpr (type == ContentType::Scene) {
76 return contentPath / kSceneRoot / filename;
77 } else if constexpr (type == ContentType::Script) {
78 return contentPath / kScriptRoot / filename;
79 } else if constexpr (type == ContentType::Shader) {
80 return contentPath / kShaderRoot / filename;
81 } else if constexpr (type == ContentType::Sprite) {
82 return contentPath / kSpriteRoot / filename;
83 } else {
84 throw std::invalid_argument("Invalid content type");
85 }
86 }
87
90 inline static void SetContentPath(const fs::path& path) {
91 sContentRoot = path;
92 }
93
96 inline static void SetEngineContentPath(const fs::path& path) {
97 sEngineContentRoot = path;
98 }
99
101 inline static void SetRuntimeDefaults() {
102 sContentRoot = fs::current_path() / "Content";
103 sEngineContentRoot = sContentRoot / "EngineContent";
104 }
105 };
