42 bool AudioEngine::Initialize() {
43 if (mInitialized) {
44 Log::Warn("AudioEngine", "Engine is already initialized.");
45 return true;
46 }
47
48 const ma_result result = ma_engine_init(nullptr, &mEngine);
49 if (result != MA_SUCCESS) {
50 Log::Error("AudioEngine", "Failed to initialize audio engine: {}", (i32)result);
51 return false;
52 }
53
54 mInitialized = true;
55 Log::Info("AudioEngine", "Engine initialized successfully");
56 return true;

59 void AudioEngine::Shutdown() {
60 if (!mInitialized) { return; }
61
62 for (const auto& pair : mSounds) {
63 if (pair.second && pair.second->loaded) { ma_sound_uninit(&pair.second->sound); }
64 }
65 mSounds.clear();
66
67 ma_engine_uninit(&mEngine);
68 mInitialized = false;
69
70 Log::Info("AudioEngine", "Engine shutdown");
71 }

73 SoundID AudioEngine::LoadSound(const fs::path& filename) {
74 if (!mInitialized) { Initialize(); }
75
76 if (!exists(filename)) {
77 Log::Error("AudioEngine", "Attempted to load missing file: {}", filename.string());
78 return kInvalidSoundID;
79 }
80
81 auto sound = std::make_unique<Sound>();
82 sound->filepath = filename.string();
83
84 const ma_result result =
85 ma_sound_init_from_file(&mEngine, filename.string().c_str(), 0, nullptr, nullptr, &sound->sound);
86
87 if (result != MA_SUCCESS) {
88 Log::Error("AudioEngine", "Failed to load sound: {} (error: {})", filename.string(), (i32)result);
89 return kInvalidSoundID;
90 }
91
92 sound->loaded = true;
93 i32 soundId = mNextSoundId++;
94 mSounds[soundId] = std::move(sound);
95
96 Log::Info("AudioEngine", "Loaded sound: {} (ID: {})", filename.string(), soundId);
97 return soundId;
98 }

100 bool AudioEngine::PlaySound(SoundID id, bool loop) {
101 auto* sound = GetSound(id);
102 if (!sound) {
103 Log::Error("AudioEngine", "Sound ID {} not found", id);
104 return false;
105 }
106
107 ma_sound_set_looping(&sound->sound, loop ? MA_TRUE : MA_FALSE);
108
109 ma_sound_seek_to_pcm_frame(&sound->sound, 0);
110
111 if (const ma_result result = ma_sound_start(&sound->sound); result != MA_SUCCESS) {
112 Log::Error("AudioEngine", "Failed to play sound ID {} (error: {})", id, CAST<i32>(result));
113 return false;
114 }
115
116 return true;
117 }

153 AudioEngine::Sound* AudioEngine::GetSound(SoundID id) {
154 const auto it = mSounds.find(id);
155 if (it == mSounds.end() || !it->second->loaded) { return nullptr; }
156 return it->second.get();
157 }
